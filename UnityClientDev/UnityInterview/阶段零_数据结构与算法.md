### 前置知识：CPU 缓存与内存层级（计算机组成原理）

理解数据结构与算法的性能表现，离不开底层内存模型：

```
CPU 寄存器  ─── 最快，~1ns，KB 级别
    ↓
L1 缓存     ─── ~1ns，~32KB/核
    ↓
L2 缓存     ─── ~5ns，~256KB/核
    ↓
L3 缓存     ─── ~15ns，~8-16MB（共享）
    ↓
主存（RAM） ─── ~100ns，GB 级别
    ↓
磁盘（SSD） ─── ~100μs（100,000ns）
```

**对算法的影响：**
- **缓存命中率**：数组（连续内存）的遍历远快于链表（离散节点），因为数组能利用 CPU 缓存行预取
- **空间局部性**：尽量使用连续内存结构（数组 → List<T>、NativeArray），避免指针跳跃（链表）
- **时间局部性**：热点数据应集中访问，避免频繁切换不同数据区域
- **Unity 场景**：Job System + IJobParallelFor 利用缓存友好性；ECS 架构的 Archetype 按组件连续存储

```csharp
// 缓存友好的访问（连续内存，快）
int[] array = new int[1000];
for (int i = 0; i < array.Length; i++)
    array[i] *= 2;

// 缓存不友好的访问（跳跃指针，慢）
LinkedListNode<int> node = list.First;
while (node != null) {
    node.Value *= 2;
    node = node.Next;
}
```

**C++ 对比**：C++ 中可手动控制缓存行对齐（`alignas(64)` 避免 false sharing）、使用 `std::vector` 代替 `std::list` 利用连续性

> 以上概念在后续所有数据结构和算法的时空复杂度分析中都会体现。面试高频问题："数组和链表的区别" 的核心其实就是缓存友好性。

### 1.1 数组与列表

#### 核心概念解析

**什么是数组？**

- 数组是一种**线性表**数据结构，用一段**连续的内存空间**存储相同类型的数据
- 特点：**随机访问**效率高 O(1)，但**插入删除**效率低 O(n)
- 内存模型：假设数组起始地址为 base，每个元素占 size 字节，则 `arr[i] = base + i * size`

```
内存示意图：
┌────┬────┬────┬────┬────┐
│ 0  │ 1  │ 2  │ 3  │ 4  │  ← 索引
├────┼────┼────┼────┼────┤
│ 10 │ 20 │ 30 │ 40 │ 50 │  ← 数据
└────┴────┴────┴────┴────┘
↑
base地址 (如 0x0010)
```

**List<T>** **与数组的区别**

- 数组大小固定，List 动态增长（自动扩容）
- 扩容机制：通常翻倍扩容（1→2→4→8...），但有容量损耗
- 适用于：需要动态大小、数据量不确定的场景

#### 数组基础


#### List 源码模拟（理解扩容）

```csharp
/// <summary>
/// 简化版动态列表实现，模拟 List<T> 的核心机制
/// </summary>
/// <typeparam name="T">元素类型</typeparam>
public class MyList<T>
{
    // 底层存储数据的数组
    private T[] items;
    
    // 当前实际存储的元素数量
    private int size;
    
    // 数组的容量（可存储的最大元素数）
    private int capacity = 4;

    /// <summary>
    /// 构造函数：初始化容量为4的空列表
    /// </summary>
    public MyList()
    {
        items = new T[capacity];
    }

    /// <summary>
    /// 获取当前元素数量（只读属性）
    /// </summary>
    public int Count => size;

    /// <summary>
    /// 在列表末尾添加元素
    /// 时间复杂度：平均O(1)，扩容时O(n)
    /// </summary>
    public void Add(T item)
    {
        // 检查是否需要扩容
        if (size >= capacity)
        {
            // 容量翻倍（常见策略）
            capacity *= 2;
            
            // 创建新的更大数组
            T[] newItems = new T[capacity];
            
            // 将旧数组元素复制到新数组
            Array.Copy(items, newItems, size);
            
            // 指向新数组
            items = newItems;
        }
        
        // 在当前末尾位置添加新元素，然后size自增
        items[size++] = item;
    }

    /// <summary>
    /// 索引器：支持像数组一样访问元素
    /// </summary>
    public T this[int index]
    {
        get
        {
            // 边界检查：索引必须在有效范围内
            if (index < 0 || index >= size)
                throw new IndexOutOfRangeException();
            return items[index];
        }
        set
        {
            // 边界检查
            if (index < 0 || index >= size)
                throw new IndexOutOfRangeException();
            items[index] = value;
        }
    }
}
```

### 动态数组详细实现

```C#
using System;

/// <summary>
/// 泛型动态数组实现（模拟 C# List<T> 的核心功能）
/// 支持动态扩容、缩容、增删改查等操作
/// </summary>
/// <typeparam name="T">元素类型</typeparam>
public class MyArrayList<T>
{
    // ========== 成员变量 ==========
    
    // 底层存储数据的数组（真正存放元素的地方）
    private T[] data;
    
    // 当前实际存储的元素数量（注意：不等于数组长度）
    private int size;
    
    // 默认初始容量（设置为1，最小化空列表的内存占用）
    private const int INIT_CAP = 1;

    // ========== 构造函数 ==========

    /// <summary>
    /// 默认构造函数：使用默认初始容量
    /// </summary>
    public MyArrayList()
    {
        // 调用有参构造函数
        this(INIT_CAP);
    }

    /// <summary>
    /// 带初始容量的构造函数
    /// </summary>
    /// <param name="initCapacity">初始容量</param>
    public MyArrayList(int initCapacity)
    {
        // 参数校验：容量不能为负数
        if (initCapacity < 0)
        {
            throw new ArgumentException("Initial capacity cannot be negative.");
        }
        
        // 创建指定容量的数组
        data = new T[initCapacity];
        
        // 初始时元素数量为0
        size = 0;
    }

    // ========== 增：添加元素 ==========

    /// <summary>
    /// 在列表末尾添加元素
    /// 时间复杂度：平均O(1)，扩容时O(n)
    /// </summary>
    public void AddLast(T e)
    {
        int cap = data.Length;
        
        // 检查是否需要扩容：当元素数量等于数组容量时
        if (size == cap)
        {
            // 扩容策略：容量翻倍
            Resize(2 * cap);
        }
        
        // 在当前末尾位置添加新元素
        data[size] = e;
        
        // 元素数量加1
        size++;
    }

    /// <summary>
    /// 在指定索引位置插入元素
    /// 时间复杂度：O(n)，需要移动元素
    /// </summary>
    /// <param name="index">插入位置</param>
    /// <param name="e">要插入的元素</param>
    public void Add(int index, T e)
    {
        // 检查索引是否有效（可以插入的位置：0到size）
        CheckPositionIndex(index);

        int cap = data.Length;
        
        // 检查是否需要扩容
        if (size == cap)
        {
            Resize(2 * cap);
        }

        // 搬移数据：将 index 及之后的元素向后移动一位
        // 从后往前移动，避免覆盖未移动的元素
        for (int i = size - 1; i >= index; i--)
        {
            data[i + 1] = data[i];
        }

        // 在指定位置插入新元素
        data[index] = e;

        // 元素数量加1
        size++;
    }

    /// <summary>
    /// 在列表头部添加元素
    /// 时间复杂度：O(n)，需要移动所有元素
    /// </summary>
    public void AddFirst(T e)
    {
        // 复用 Add 方法，在索引0位置插入
        Add(0, e);
    }

    // ========== 删：删除元素 ==========

    /// <summary>
    /// 删除并返回最后一个元素
    /// 时间复杂度：O(1)
    /// </summary>
    public T RemoveLast()
    {
        // 检查列表是否为空
        if (size == 0)
        {
            throw new InvalidOperationException("List is empty.");
        }
        
        int cap = data.Length;
        
        // 缩容策略：当元素数量 <= 容量的1/4时，将容量减半
        // 防止频繁扩容缩容带来的震荡
        if (size == cap / 4)
        {
            Resize(cap / 2);
        }

        // 保存要删除的元素值
        T deletedVal = data[size - 1];
        
        // 将最后一个元素置为默认值（对于引用类型即null）
        // 这一步很重要，防止内存泄漏/对象被意外持有
        data[size - 1] = default(T); 
        
        // 元素数量减1
        size--;

        return deletedVal;
    }

    /// <summary>
    /// 删除指定索引位置的元素并返回
    /// 时间复杂度：O(n)，需要移动元素
    /// </summary>
    public T Remove(int index)
    {
        // 检查索引是否有效（必须是已存在元素的索引）
        CheckElementIndex(index);

        int cap = data.Length;
        
        // 检查是否需要缩容
        if (size == cap / 4)
        {
            Resize(cap / 2);
        }

        // 保存要删除的元素值
        T deletedVal = data[index];

        // 搬移数据：将 index+1 及之后的元素向前移动一位
        for (int i = index + 1; i < size; i++)
        {
            data[i - 1] = data[i];
        }

        // 将最后一个位置置为默认值
        data[size - 1] = default(T);
        
        // 元素数量减1
        size--;

        return deletedVal;
    }

    /// <summary>
    /// 删除并返回第一个元素
    /// 时间复杂度：O(n)，需要移动所有元素
    /// </summary>
    public T RemoveFirst()
    {
        // 复用 Remove 方法
        return Remove(0);
    }

    // ========== 查：获取元素 ==========

    /// <summary>
    /// 获取指定索引位置的元素
    /// 时间复杂度：O(1)
    /// </summary>
    public T Get(int index)
    {
        // 检查索引是否有效
        CheckElementIndex(index);

        return data[index];
    }

    // ========== 改：修改元素 ==========

    /// <summary>
    /// 修改指定索引位置的元素，返回旧值
    /// 时间复杂度：O(1)
    /// </summary>
    public T Set(int index, T element)
    {
        // 检查索引是否有效
        CheckElementIndex(index);
        
        // 保存旧值
        T oldVal = data[index];
        
        // 修改为新值
        data[index] = element;
        
        return oldVal;
    }

    // ========== 工具方法 ==========

    /// <summary>
    /// 获取当前元素数量
    /// </summary>
    public int Size()
    {
        return size;
    }

    /// <summary>
    /// 判断列表是否为空
    /// </summary>
    public bool IsEmpty()
    {
        return size == 0;
    }

    /// <summary>
    /// 调整底层数组容量
    /// 时间复杂度：O(n)，需要复制所有元素
    /// </summary>
    private void Resize(int newCap)
    {
        // 创建新容量的数组
        T[] temp = new T[newCap];

        // 将旧数组元素复制到新数组
        for (int i = 0; i < size; i++)
        {
            temp[i] = data[i];
        }

        // 指向新数组
        data = temp;
    }

    // ========== 边界检查辅助方法 ==========

    /// <summary>
    /// 判断索引是否为有效的元素索引（0 <= index < size）
    /// 用于 Get、Set、Remove 等操作
    /// </summary>
    private bool IsElementIndex(int index)
    {
        return index >= 0 && index < size;
    }

    /// <summary>
    /// 判断索引是否为有效的插入位置（0 <= index <= size）
    /// 用于 Add 操作
    /// </summary>
    private bool IsPositionIndex(int index)
    {
        return index >= 0 && index <= size;
    }

    /// <summary>
    /// 检查元素索引是否有效，无效则抛出异常
    /// </summary>
    private void CheckElementIndex(int index)
    {
        if (!IsElementIndex(index))
        {
            throw new ArgumentOutOfRangeException(nameof(index), $"Index: {index}, Size: {size}");
        }
    }

    /// <summary>
    /// 检查插入位置是否有效，无效则抛出异常
    /// </summary>
    private void CheckPositionIndex(int index)
    {
        if (!IsPositionIndex(index))
        {
            throw new ArgumentOutOfRangeException(nameof(index), $"Index: {index}, Size: {size}");
        }
    }

    // ========== 调试辅助方法 ==========

    /// <summary>
    /// 打印列表内容（用于调试）
    /// </summary>
    public void Display()
    {
        Console.WriteLine($"size = {size} cap = {data.Length}");
        Console.Write("[");
        for (int i = 0; i < size; i++)
        {
            Console.Write(data[i]);
            if (i < size - 1) Console.Write(", ");
        }
        Console.WriteLine("]");
    }

    // ========== 测试示例 ==========

    public static void Main(string[] args)
    {
        // 创建初始容量为3的动态数组
        MyArrayList<int> arr = new MyArrayList<int>(3);

        // 添加5个元素（会触发扩容：3 -> 6）
        for (int i = 1; i <= 5; i++)
        {
            arr.AddLast(i);
        }

        // 删除索引3的元素（值为4）
        arr.Remove(3);
        
        // 在索引1位置插入9
        arr.Add(1, 9);
        
        // 在头部添加100
        arr.AddFirst(100);
        
        // 删除最后一个元素（值为5）
        int val = arr.RemoveLast();

        // 遍历输出结果：100, 1, 9, 2, 3
        for (int i = 0; i < arr.Size(); i++)
        {
            Console.WriteLine(arr.Get(i));
        }
    }
}
```

### 1.2 链表

#### 核心概念解析

**什么是链表？**

- 链表是一种**线性表**数据结构，但不像数组那样需要连续内存
- 由**节点（Node）组成，每个节点包含：数据域**和指针域
- 特点：**插入删除**效率高 O(1)，但**随机访问**效率低 O(n)

```
单链表结构示意图：
┌─────┐    ┌─────┐    ┌─────┐    ┌─────┐
│Data │    │Data │    │Data │    │Data │
│ 10  │ →  │ 20  │ →  │ 30  │ →  │ 40  │ → null
└─────┘    └─────┘    └─────┘    └─────┘
  ↑                            ↑
 head                        tail
```

**链表 vs 数组对比**

| 特性                                 | 数组        | 链表      |
| ---------------------------------- | --------- | ------- |
| 内存分布                               | 连续        | 分散      |
| 访问方式                               | 随机访问 O(1) | 遍历 O(n) |
| 插入/删除                              | O(n)      | O(1)\*  |
| 空间利用率                              | 高         | 低（需存指针） |
| \*注：链表插入删除 O(1)指找到位置后，但找到位置仍需 O(n) | <br />    | <br />  |

**链表类型**

1. **单向链表**：只能从头遍历到尾
2. **双向链表**：每个节点有 prev 和 next 指针
3. **循环链表**：尾节点指向头节点

#### 单链表实现

```csharp
/// <summary>
/// 单链表节点定义
/// </summary>
/// <typeparam name="T">节点存储的数据类型</typeparam>
public class ListNode<T>
{
    /// <summary>
    /// 数据域：存储节点的值
    /// </summary>
    public T Value { get; set; }
    
    /// <summary>
    /// 指针域：指向下一个节点
    /// </summary>
    public ListNode<T> Next { get; set; }

    /// <summary>
    /// 构造函数
    /// </summary>
    public ListNode(T value)
    {
        Value = value;
        // Next 默认值为 null
    }
}

/// <summary>
/// 单链表实现
/// </summary>
/// <typeparam name="T">元素类型</typeparam>
public class LinkedList<T>
{
    // 头节点引用（链表的入口）
    private ListNode<T> head;
    
    // 链表长度（元素数量）
    private int count;

    /// <summary>
    /// 获取链表长度
    /// </summary>
    public int Count => count;

    // ========== 插入操作 ==========

    /// <summary>
    /// 在链表头部插入元素
    /// 时间复杂度：O(1)
    /// </summary>
    public void AddFirst(T value)
    {
        // 创建新节点
        ListNode<T> newNode = new ListNode<T>(value);
        
        // 新节点的 Next 指向原来的头节点
        newNode.Next = head;
        
        // 更新头节点为新节点
        head = newNode;
        
        // 长度加1
        count++;
    }

    /// <summary>
    /// 在链表尾部插入元素
    /// 时间复杂度：O(n)，需要遍历到尾部
    /// </summary>
    public void AddLast(T value)
    {
        // 创建新节点
        ListNode<T> newNode = new ListNode<T>(value);

        // 如果链表为空，新节点就是头节点
        if (head == null)
        {
            head = newNode;
        }
        else
        {
            // 遍历到最后一个节点
            ListNode<T> current = head;
            while (current.Next != null)
            {
                current = current.Next;
            }
            // 将最后一个节点的 Next 指向新节点
            current.Next = newNode;
        }
        
        // 长度加1
        count++;
    }

    // ========== 删除操作 ==========

    /// <summary>
    /// 删除第一个匹配的节点
    /// 时间复杂度：O(n)
    /// </summary>
    /// <returns>删除成功返回true，未找到返回false</returns>
    public bool Remove(T value)
    {
        // 空链表直接返回false
        if (head == null) return false;

        // 如果要删除的是头节点
        if (head.Value.Equals(value))
        {
            head = head.Next;
            count--;
            return true;
        }

        // 查找要删除节点的前一个节点
        ListNode<T> current = head;
        while (current.Next != null)
        {
            if (current.Next.Value.Equals(value))
            {
                // 跳过被删除的节点
                current.Next = current.Next.Next;
                count--;
                return true;
            }
            current = current.Next;
        }
        
        // 未找到匹配元素
        return false;
    }

    // ========== 遍历操作 ==========

    /// <summary>
    /// 遍历链表，对每个元素执行指定操作
    /// </summary>
    public void ForEach(Action<T> action)
    {
        ListNode<T> current = head;
        while (current != null)
        {
            action(current.Value);
            current = current.Next;
        }
    }
}
```

### 1.3 栈

#### 核心概念解析

**什么是栈？**

- 栈是一种\*\*后进先出（LIFO, Last In First Out）\*\*的线性表
- 就像一叠盘子：最后放上去的盘子最先被拿走
- 基本操作：**Push（压栈）和**Pop（弹栈）

```
栈操作示意图：
Push(1)  Push(2)  Push(3)  Pop()    Pop()
┌───┐    ┌───┐    ┌───┐    ┌───┐    ┌───┐
│   │    │   │    │ 3  │    │   │    │   │
├───┤    ├───┤    ├───┤    ├───┤    ├───┤
│   │    │ 1  │    │ 2  │    │ 2  │    │ 1  │
├───┤    ├───┤    ├───┤    ├───┤    ├───┤
│ 1 │    │   │    │ 1  │    │ 1  │    │   │
└───┘    └───┘    └───┘    └───┘    └───┘
```

**栈的核心特性**

- **只能在栈顶操作**：入栈、出栈、查看栈顶
- **逆序特性**：数据入栈顺序与出栈顺序相反
- **递归实现基础**：函数调用栈就是用栈实现的

**栈的存储结构**

- 可以用**数组**实现（顺序栈）
- 也可以用**链表**实现（链式栈）

**栈的应用场景**

- 函数调用栈（递归实现基础）
- 括号匹配、表达式求值
- 撤销操作（编辑器的 Ctrl+Z）
- 深度优先搜索（DFS）

#### 栈实现

```csharp
/// <summary>
/// 基于数组实现的栈（顺序栈）
/// 遵循后进先出（LIFO）原则
/// </summary>
/// <typeparam name="T">元素类型</typeparam>
public class Stack<T>
{
    // 底层存储数据的数组
    private T[] items;
    
    // 栈顶指针：指向栈顶元素的索引（-1表示空栈）
    private int top;
    
    // 数组容量
    private int capacity = 4;

    /// <summary>
    /// 构造函数：创建容量为4的空栈
    /// </summary>
    public Stack()
    {
        items = new T[capacity];
        top = -1;  // 初始为空栈
    }

    /// <summary>
    /// 获取栈中元素数量
    /// </summary>
    public int Count => top + 1;

    /// <summary>
    /// 判断栈是否为空
    /// </summary>
    public bool IsEmpty => top < 0;

    // ========== 核心操作 ==========

    /// <summary>
    /// 入栈：将元素压入栈顶
    /// 时间复杂度：平均O(1)，扩容时O(n)
    /// </summary>
    public void Push(T item)
    {
        // 检查是否需要扩容
        if (top >= capacity - 1)
        {
            // 容量翻倍
            capacity *= 2;
            T[] newItems = new T[capacity];
            // 复制原数组元素
            Array.Copy(items, newItems, items.Length);
            items = newItems;
        }
        
        // top自增后赋值（先++top，再赋值）
        items[++top] = item;
    }

    /// <summary>
    /// 出栈：移除并返回栈顶元素
    /// 时间复杂度：O(1)
    /// </summary>
    /// <exception cref="InvalidOperationException">栈为空时抛出</exception>
    public T Pop()
    {
        if (top < 0)
            throw new InvalidOperationException("Stack is empty");
        
        // 返回栈顶元素，然后top自减（先返回items[top]，再--top）
        return items[top--];
    }

    /// <summary>
    /// 查看栈顶元素（不移除）
    /// 时间复杂度：O(1)
    /// </summary>
    /// <exception cref="InvalidOperationException">栈为空时抛出</exception>
    public T Peek()
    {
        if (top < 0)
            throw new InvalidOperationException("Stack is empty");
        return items[top];
    }
}
```

```csharp
// ========== 应用场景1：撤销操作（Undo/Redo）==========
public class UndoManager
{
    // 撤销栈：存储待撤销的操作
    private Stack<Action> undoStack = new Stack<Action>();
    
    // 重做栈：存储待重做的操作
    private Stack<Action> redoStack = new Stack<Action>();

    /// <summary>
    /// 执行操作并记录到撤销栈
    /// </summary>
    public void Do(Action action)
    {
        // 执行操作
        action();
        
        // 将反向操作压入撤销栈
        undoStack.Push(() =>
        {
            action();  // 再次执行即为撤销（假设操作是可逆的）
            redoStack.Push(action);  // 同时记录到重做栈
        });
    }

    /// <summary>
    /// 撤销上一次操作
    /// </summary>
    public void Undo()
    {
        if (undoStack.Count > 0)
        {
            Action action = undoStack.Pop();
            action();  // 执行撤销操作
        }
    }
}

// ========== 应用场景2：括号匹配验证 ==========
public bool IsValidBracket(string s)
{
    Stack<char> stack = new Stack<char>();
    
    foreach (char c in s)
    {
        // 如果是左括号，压入栈
        if (c == '(' || c == '[' || c == '{')
        {
            stack.Push(c);
        }
        // 如果是右括号
        else if (c == ')' || c == ']' || c == '}')
        {
            // 如果栈为空，说明右括号多余
            if (stack.Count == 0) return false;
            
            // 弹出栈顶左括号，检查是否匹配
            char top = stack.Pop();
            if ((c == ')' && top != '(') ||
                (c == ']' && top != '[') ||
                (c == '}' && top != '{'))
            {
                return false;  // 括号不匹配
            }
        }
    }
    
    // 最终栈必须为空（所有左括号都已匹配）
    return stack.Count == 0;
}

// ========== 应用场景3：逆波兰表达式求值 ==========
public int EvalRPN(string[] tokens)
{
    Stack<int> stack = new Stack<int>();
    
    foreach (string token in tokens)
    {
        // 如果是数字，压入栈
        if (int.TryParse(token, out int num))
        {
            stack.Push(num);
        }
        // 如果是运算符
        else
        {
            // 弹出两个操作数（注意顺序：后弹出的是第一个操作数）
            int b = stack.Pop();  // 第二个操作数
            int a = stack.Pop();  // 第一个操作数
            
            // 根据运算符计算
            int result = token switch
            {
                "+" => a + b,
                "-" => a - b,
                "*" => a * b,
                "/" => a / b,
                _ => 0
            };
            
            // 将结果压入栈
            stack.Push(result);
        }
    }
    
    // 最终栈顶就是结果
    return stack.Pop();
}
```

### 1.4 队列

#### 核心概念解析

**什么是队列？**

- 队列是一种\*\*先进先出（FIFO, First In First Out）\*\*的线性表
- 就像排队买票：先到的人先买到
- 基本操作：**Enqueue（入队）和**Dequeue（出队）

```
队列操作示意图：
Enqueue(1) Enqueue(2) Enqueue(3) Dequeue()  Dequeue()
┌───┬───┬───┬───┐    ┌───┬───┬───┬───┐    ┌───┬───┬───┬───┐
│   │   │   │   │    │ 1 │ 2 │ 3 │   │    │   │ 2 │ 3 │   │
└───┴───┴───┴───┘    └───┴───┴───┴───┘    └───┴───┴───┴───┘
  ↑               ↑        ↑         ↑           ↑
 front          rear     front     rear        front

Dequeue() 返回 1  →  2 成为新的 front
```

**队列的核心特性**

- **一端入队，一端出队**
- 队头（Front）：只能出队
- 队尾（Rear/Back）：只能入队

**队列的类型**

1. **普通队列**：基本 FIFO
2. **循环队列**：用数组实现时首尾相接，避免假溢出
3. **双端队列（Deque）**：两端都可以入队出队
4. **优先队列（Priority Queue）**：按优先级出队

**队列的应用场景**

- 任务调度（CPU 任务队列）
- 广度优先搜索（BFS）
- 消息队列、请求缓冲
- 打印队列

#### 队列实现

```csharp
/// <summary>
/// 基于数组实现的队列（顺序队列）
/// 遵循先进先出（FIFO）原则
/// </summary>
/// <typeparam name="T">元素类型</typeparam>
public class Queue<T>
{
    // 底层存储数据的数组
    private T[] items;
    
    // 队头指针：指向队头元素的索引
    private int head;
    
    // 队尾指针：指向下一个可插入位置的索引
    private int tail;
    
    // 当前元素数量
    private int count;
    
    // 数组容量
    private int capacity = 4;

    /// <summary>
    /// 构造函数：创建容量为4的空队列
    /// </summary>
    public Queue()
    {
        items = new T[capacity];
        // head 和 tail 默认初始化为0
    }

    /// <summary>
    /// 获取队列中元素数量
    /// </summary>
    public int Count => count;

    /// <summary>
    /// 判断队列是否为空
    /// </summary>
    public bool IsEmpty => count == 0;

    // ========== 核心操作 ==========

    /// <summary>
    /// 入队：将元素添加到队尾
    /// 时间复杂度：平均O(1)，扩容时O(n)
    /// </summary>
    public void Enqueue(T item)
    {
        // 检查是否需要扩容
        if (count >= capacity)
        {
            // 容量翻倍
            capacity *= 2;
            T[] newItems = new T[capacity];
            
            // 将有效元素复制到新数组（从head开始）
            Array.Copy(items, head, newItems, 0, count);
            
            items = newItems;
            head = 0;        // 重置队头为0
            tail = count;    // 队尾指向当前元素数量位置
        }
        
        // 在队尾添加元素，然后tail自增
        items[tail++] = item;
        count++;
    }

    /// <summary>
    /// 出队：移除并返回队头元素
    /// 时间复杂度：O(1)
    /// </summary>
    /// <exception cref="InvalidOperationException">队列为空时抛出</exception>
    public T Dequeue()
    {
        if (count == 0)
            throw new InvalidOperationException("Queue is empty");
        
        // 获取队头元素，然后head自增
        T item = items[head++];
        count--;
        
        return item;
    }

    /// <summary>
    /// 查看队头元素（不移除）
    /// 时间复杂度：O(1)
    /// </summary>
    /// <exception cref="InvalidOperationException">队列为空时抛出</exception>
    public T Peek()
    {
        if (count == 0)
            throw new InvalidOperationException("Queue is empty");
        return items[head];
    }
}
```

```csharp
// ========== 应用场景1：任务队列（游戏开发常用）==========
public class TaskQueue
{
    // 存储待执行的任务
    private Queue<Action> tasks = new Queue<Action>();

    /// <summary>
    /// 将任务加入队列
    /// </summary>
    public void Enqueue(Action task)
    {
        tasks.Enqueue(task);
    }

    /// <summary>
    /// 每帧更新时执行一个任务
    /// </summary>
    public void Update()
    {

        {

        }
    }
}

// ========== 应用场景2：广度优先搜索（BFS）==========
public void BFS(TreeNode root)
{
    // 边界检查
    if (root == null) return;

    // 使用队列存储待访问的节点
    Queue<TreeNode> queue = new Queue<TreeNode>();
    queue.Enqueue(root);  // 从根节点开始

    // 当队列不为空时继续
    while (queue.Count > 0)
    {
        // 出队一个节点
        TreeNode node = queue.Dequeue();
        
        // 访问当前节点（例如打印）
        Console.WriteLine(node.Value);

        // 将左右子节点入队（如果存在）
        if (node.Left != null)
            queue.Enqueue(node.Left);
        if (node.Right != null)
            queue.Enqueue(node.Right);
    }
}
```

### 1.5 哈希表

#### 核心概念解析

**什么是哈希表？**

- 哈希表是一种通过**哈希函数**将\*\*键（Key）**映射到**值（Value）\*\*的数据结构
- 核心思想：**用空间换时间**，查找时间接近 O(1)

```
哈希表工作原理：
┌─────────────┐      哈希函数      ┌─────────────┐
│ Key: "apple"│ ──→  hash("apple") ──→  Index: 3  │
│ Value: 100  │                        └─────┬─────┘
└─────────────┘                              │
                                          ┌─────▼─────┐
                                          │  数组[3]  │
                                          │ key:value │
                                          └───────────┘
```

**哈希函数**

- 将任意长度的输入转换为固定长度的输出
- 常见哈希函数：直接地址法、除留余数法、平方取中法
- 优秀的哈希函数应具备：**均匀分布**、**碰撞少**、**计算快**

**哈希冲突（Hash Collision）**

- 不同 key 通过哈希函数得到相同的索引
- 解决冲突的方法：
  1. **链地址法（Separate Chaining）**：冲突的元素用链表连接
  2. **开放地址法（Open Addressing）**：找下一个空位置
     - 线性探测：index+1, index+2...
     - 二次探测：index+1², index+2²...
     - 双重哈希：再用另一个哈希函数

```
链地址法示意图：
数组下标:  0      1           2         3         4
        ┌───┐ ┌───┐     ┌───┐    ┌───┐    ┌───┐
        │   │ │   │     │   │    │ ●─┼────┼─● │
        └───┘ └───┘     └───┘    └───┘    └───┘
                              (key1,val1) (key2,val2)
```

**哈希表的性能**

- 理想情况：查找、插入、删除都是 **O(1)**
- 最坏情况（大量冲突）：**O(n)**
- 装填因子（Load Factor）= 元素数量 / 容量，>0.7 时建议扩容

**哈希表的应用**

- 字典、映射
- 缓存（LRU Cache）
- 唯一性检验（去重）
- 编译器符号表

```csharp
/// <summary>
/// 简单哈希表实现（使用链地址法解决冲突）
/// </summary>
/// <typeparam name="K">键类型</typeparam>
/// <typeparam name="V">值类型</typeparam>
public class HashMap<K, V>
{
    /// <summary>
    /// 键值对节点（用于链地址法存储冲突元素）
    /// </summary>
    private class Entry
    {
        public K Key;           // 键
        public V Value;         // 值
        public int HashCode;    // 哈希码（缓存）
        public Entry Next;      // 下一个节点（解决冲突用）
    }

    // 哈希表的底层数组（桶数组）
    private Entry[] table;
    
    // 数组容量（桶的数量）
    private int capacity = 16;
    
    // 当前元素数量
    private int size;

    /// <summary>
    /// 构造函数：创建容量为16的空哈希表
    /// </summary>
    public HashMap()
    {
        table = new Entry[capacity];
    }

    /// <summary>
    /// 根据键计算数组索引
    /// </summary>
    private int GetIndex(K key)
    {
        // 获取哈希码，并通过与运算确保非负
        int hash = key.GetHashCode() & 0x7FFFFFFF;
        // 使用取模运算得到索引
        return hash % capacity;
    }

    /// <summary>
    /// 添加或更新键值对
    /// 时间复杂度：平均O(1)，最坏O(n)（哈希冲突严重时）
    /// </summary>
    public void Put(K key, V value)
    {
        // 1. 计算索引
        int index = GetIndex(key);
        
        // 2. 遍历链表，查找是否已存在该键
        Entry current = table[index];
        while (current != null)
        {
            // 如果找到相同的键，更新值
            if (current.Key.Equals(key))
            {
                current.Value = value;
                return;
            }
            current = current.Next;
        }

        // 3. 键不存在，创建新节点
        Entry newEntry = new Entry
        {
            Key = key,
            Value = value,
            HashCode = GetIndex(key)
        };
        
        // 4. 头插法插入链表
        newEntry.Next = table[index];
        table[index] = newEntry;
        
        // 5. 元素数量加1
        size++;
    }

    /// <summary>
    /// 根据键获取值
    /// 时间复杂度：平均O(1)，最坏O(n)
    /// </summary>
    public V Get(K key)
    {
        // 1. 计算索引
        int index = GetIndex(key);
        
        // 2. 遍历链表查找
        Entry current = table[index];
        while (current != null)
        {
            if (current.Key.Equals(key))
                return current.Value;
            current = current.Next;
        }
        
        // 3. 未找到，返回默认值
        return default;
    }
}
```

### 2.1 二叉树

#### 核心概念解析

**什么是树？**

- 树是一种**非线性**数据结构，由**节点**组成，每个节点有零个或多个子节点
- 树中有一个特殊节点称为**根节点（Root）**，没有父节点

**二叉树基本概念**

- **二叉树**：每个节点最多有两个子节点的树（左孩子、右孩子）
- **满二叉树**：所有节点都有两个子节点，且所有叶子节点在同一层
- **完全二叉树**：除最后一层外，每层节点数达到最大，最后一层从左到右排列

```
二叉树示意图：

          根节点
            │
          ┌─┴─┐
          │ 1 │
          └───┘
         ↙   ↘
      ┌─┴─┐ ┌─┴─┐
      │ 2 │ │ 3 │
      └───┘ └───┘
     ↙   ↘     ↘
   ┌─┴─┐ ┌─┴─┐ ┌─┴─┐
   │ 4 │ │ 5 │ │ 6 │
   └───┘ └───┘ └───┘
```

**树的术语**

- **节点深度（Depth）**：从根节点到该节点的路径长度（根深度=0）
- **节点高度（Height）**：从该节点到最深叶子节点的距离（叶子高度=0）
- **树的深度/高度**：根节点的深度/高度
- **父节点（Parent）**：有子节点的节点
- **子节点（Children）**：某个节点下的节点
- **叶子节点（Leaf）**：没有子节点的节点
- **度（Degree）**：节点的子节点数量

**二叉树的存储结构**

1. **顺序存储**（数组）：适合完全二叉树，索引 i 的左孩子在 2i+1，右孩子在 2i+2
2. **链式存储**（指针）：每个节点含 data、left、right 三个字段

**二叉树的应用**

- 二叉搜索树（BST）：高效查找
- 堆：优先队列、Top K 问题
- 哈夫曼树：数据压缩
- 决策树：机器学习

#### 二叉树遍历

```csharp
/// <summary>
/// 二叉树遍历实现
/// </summary>
public class BinaryTreeTraversal
{
    // ========== 深度优先遍历（DFS） ==========

    /// <summary>
    /// 前序遍历：根 -> 左 -> 右
    /// 应用：复制二叉树、序列化
    /// </summary>
    public void PreOrder(TreeNode<int> root)
    {
        // 递归终止条件
        if (root == null) return;
        
        // 1. 访问根节点
        Console.Write(root.Value + " ");
        
        // 2. 递归遍历左子树
        PreOrder(root.Left);
        
        // 3. 递归遍历右子树
        PreOrder(root.Right);
    }

    /// <summary>
    /// 中序遍历：左 -> 根 -> 右
    /// 应用：二叉搜索树的有序输出
    /// </summary>
    public void InOrder(TreeNode<int> root)
    {
        if (root == null) return;
        
        // 1. 递归遍历左子树
        InOrder(root.Left);
        
        // 2. 访问根节点
        Console.Write(root.Value + " ");
        
        // 3. 递归遍历右子树
        InOrder(root.Right);
    }

    /// <summary>
    /// 后序遍历：左 -> 右 -> 根
    /// 应用：删除二叉树、计算子树和
    /// </summary>
    public void PostOrder(TreeNode<int> root)
    {
        if (root == null) return;
        
        // 1. 递归遍历左子树
        PostOrder(root.Left);
        
        // 2. 递归遍历右子树
        PostOrder(root.Right);
        
        // 3. 访问根节点
        Console.Write(root.Value + " ");
    }

    // ========== 广度优先遍历（BFS） ==========

    /// <summary>
    /// 层序遍历：按层次从上到下、从左到右访问
    /// 应用：树的高度计算、层序输出
    /// </summary>
    public void LevelOrder(TreeNode<int> root)
    {
        // 边界检查
        if (root == null) return;

        // 使用队列存储待访问的节点
        Queue<TreeNode<int>> queue = new Queue<TreeNode<int>>();
        queue.Enqueue(root);

        // 当队列不为空时继续
        while (queue.Count > 0)
        {
            // 出队一个节点
            TreeNode<int> node = queue.Dequeue();
            
            // 访问当前节点
            Console.Write(node.Value + " ");

            // 将左右子节点入队（如果存在）
            if (node.Left != null)
                queue.Enqueue(node.Left);
            if (node.Right != null)
                queue.Enqueue(node.Right);
        }
    }
}
```

### 写法一：最简单，但无法记录深度

**实现思路：**

1. 初始化队列，将根节点入队。
2. 当队列不为空时，取出队头节点 `cur`。
3. 访问 `cur`。
4. 如果 `cur` 有左/右子节点，将其加入队列。
5. **缺点**：你只知道节点被访问了，但不知道它属于第几层。

~~~C#
using System;
using System.Collections.Generic;

public void LevelOrderTraverseV1(TreeNode root) {
    if (root == null) return;
    
    Queue<TreeNode> q = new Queue<TreeNode>();
    q.Enqueue(root);
    
    while (q.Count > 0) {
        TreeNode cur = q.Dequeue();
        // 访问 cur 节点
        Console.WriteLine(cur.val);

        // 把 cur 的左右子节点加入队列
        if (cur.left != null) {
            q.Enqueue(cur.left);
        }
        if (cur.right != null) {
            q.Enqueue(cur.right);
        }
    }
}
~~~

### 写法二：最常用，可记录深度

**实现思路：**

1. 在写法一的基础上，增加一个 `depth` 变量记录当前层数。
2. **关键点**：在进入每一层的循环前，先获取当前队列的大小 `sz`。这个 `sz` 就是**当前层节点的个数**。
3. 使用一个 `for` 循环（或 `while(sz-- > 0)`）精确处理完这 `sz` 个节点。
4. 在这 `sz` 个节点的处理过程中，新加入队列的是下一层的节点，不会影响当前层的计数。
5. 内层循环结束后，`depth++`，进入下一层。

~~~C#
public void LevelOrderTraverseV2(TreeNode root) {
    if (root == null) return;
    
    Queue<TreeNode> q = new Queue<TreeNode>();
    q.Enqueue(root);
    // 记录当前遍历到的层数（根节点视为第 1 层）
    int depth = 1;

    while (q.Count > 0) {
        int sz = q.Count; // 关键：保存当前层的节点数量
        for (int i = 0; i < sz; i++) {
            TreeNode cur = q.Dequeue();
            // 访问 cur 节点，同时知道它所在的层数
            Console.WriteLine($"depth = {depth}, val = {cur.val}");

            // 把 cur 的左右子节点加入队列
            if (cur.left != null) {
                q.Enqueue(cur.left);
            }
            if (cur.right != null) {
                q.Enqueue(cur.right);
            }
        }
        depth++; // 一层处理完毕，深度加 1
    }
}
~~~

### 写法三：最灵活，常用于复杂场景（带状态）

**实现思路：**

1. 当每个节点需要携带额外信息（如路径权重、特定状态）时，写法二的单一 `depth` 变量不够用。
2. 定义一个 `State` 类，包含节点引用和该节点对应的状态（如深度 `depth`）。
3. 队列中存储的是 `State` 对象，而不是单纯的 `TreeNode`。
4. 每次出队时，直接获取该节点对应的状态。
5. 子节点入队时，根据父节点的状态计算出新状态（例如 `parent.depth + 1`）。
6. **优点**：每个节点独立维护自己的状态，适用于边有权重或非均匀深度的图/树遍历。

~~~C#
// 定义状态类
public class State {
    public TreeNode node;
    public int depth;

    public State(TreeNode node, int depth) {
        this.node = node;
        this.depth = depth;
    }
}

public void LevelOrderTraverseV3(TreeNode root) {
    if (root == null) return;
    
    Queue<State> q = new Queue<State>();
    // 根节点的路径权重和/深度是 1
    q.Enqueue(new State(root, 1));

    while (q.Count > 0) {
        State cur = q.Dequeue();
        // 访问 cur 节点，同时知道它的路径权重和/深度
        Console.WriteLine($"depth = {cur.depth}, val = {cur.node.val}");

        // 把 cur 的左右子节点加入队列，并继承/计算新状态
        if (cur.node.left != null) {
            q.Enqueue(new State(cur.node.left, cur.depth + 1));
        }
        if (cur.node.right != null) {
            q.Enqueue(new State(cur.node.right, cur.depth + 1));
        }
    }
}
~~~

### 2.2 二叉搜索树

#### BST 实现

```csharp
/// <summary>
/// 二叉搜索树（Binary Search Tree）实现
/// 特性：左子树所有节点值 < 根节点值 < 右子树所有节点值
/// </summary>
/// <typeparam name="T">元素类型，必须实现 IComparable 接口</typeparam>
public class BST<T> where T : IComparable
{
    /// <summary>
    /// BST 节点定义
    /// </summary>
    private class Node
    {
        public T Value;      // 节点值
        public Node Left;    // 左子节点
        public Node Right;   // 右子节点
        
        public Node(T value) { Value = value; }
    }

    // 根节点
    private Node root;

    // ========== 插入操作 ==========

    /// <summary>
    /// 向BST中插入元素
    /// 时间复杂度：平均O(log n)，最坏O(n)
    /// </summary>
    public void Insert(T value)
    {
        root = InsertRecursive(root, value);
    }

    /// <summary>
    /// 递归插入辅助方法
    /// </summary>
    private Node InsertRecursive(Node node, T value)
    {
        // 递归终止：找到空位置，创建新节点
        if (node == null)
            return new Node(value);

        // 如果值小于当前节点，插入左子树
        if (value.CompareTo(node.Value) < 0)
            node.Left = InsertRecursive(node.Left, value);
        // 如果值大于当前节点，插入右子树
        else if (value.CompareTo(node.Value) > 0)
            node.Right = InsertRecursive(node.Right, value);
        // 如果值相等，不插入（BST通常不允许重复值）

        return node;
    }

    // ========== 搜索操作 ==========

    /// <summary>
    /// 在BST中搜索元素
    /// 时间复杂度：平均O(log n)，最坏O(n)
    /// </summary>
    public Node Search(T value)
    {
        return SearchRecursive(root, value);
    }

    /// <summary>
    /// 递归搜索辅助方法
    /// </summary>
    private Node SearchRecursive(Node node, T value)
    {
        // 终止条件：节点为空或找到匹配节点
        if (node == null || node.Value.CompareTo(value) == 0)
            return node;

        // 如果值小于当前节点，搜索左子树
        if (value.CompareTo(node.Value) < 0)
            return SearchRecursive(node.Left, value);
        // 如果值大于当前节点，搜索右子树
        return SearchRecursive(node.Right, value);
    }

    // ========== 删除操作 ==========

    /// <summary>
    /// 从BST中删除元素
    /// 时间复杂度：平均O(log n)，最坏O(n)
    /// </summary>
    public void Delete(T value)
    {
        root = DeleteRecursive(root, value);
    }

    /// <summary>
    /// 递归删除辅助方法
    /// 删除分为三种情况：
    /// 1. 叶子节点：直接删除
    /// 2. 只有一个子节点：用子节点替代
    /// 3. 有两个子节点：用后继（右子树最小值）替代
    /// </summary>
    private Node DeleteRecursive(Node node, T value)
    {
        // 未找到要删除的节点
        if (node == null) return null;

        // 查找要删除的节点
        if (value.CompareTo(node.Value) < 0)
        {
            node.Left = DeleteRecursive(node.Left, value);
        }
        else if (value.CompareTo(node.Value) > 0)
        {
            node.Right = DeleteRecursive(node.Right, value);
        }
        else
        {
            // 找到要删除的节点
            
            // 情况1：只有右子节点或无子节点
            if (node.Left == null) return node.Right;
            
            // 情况2：只有左子节点
            if (node.Right == null) return node.Left;

            // 情况3：有两个子节点
            // 找到右子树中的最小值（后继）
            node.Value = MinValue(node.Right);
            // 删除后继节点
            node.Right = DeleteRecursive(node.Right, node.Value);
        }
        return node;
    }

    /// <summary>
    /// 查找子树中的最小值（最左节点）
    /// </summary>
    private T MinValue(Node node)
    {
        T minValue = node.Value;
        while (node.Left != null)
        {
            minValue = node.Left.Value;
            node = node.Left;
        }
        return minValue;
    }
}
```

### 2.3 图

#### 图的表示

```csharp
/// <summary>
/// 图的邻接表表示
/// 适用于稀疏图（边数较少的图）
/// </summary>
public class Graph
{
    // 邻接表：键为顶点，值为该顶点的所有邻居列表
    private Dictionary<int, List<int>> adjList;

    /// <summary>
    /// 构造函数：创建空图
    /// </summary>
    public Graph()
    {
        adjList = new Dictionary<int, List<int>>();
    }

    /// <summary>
    /// 向无向图中添加边
    /// 无向图中边是双向的：v1-v2 和 v2-v1
    /// </summary>
    public void AddEdge(int v1, int v2)
    {
        // 如果 v1 不在邻接表中，添加
        if (!adjList.ContainsKey(v1))
            adjList[v1] = new List<int>();
        
        // 如果 v2 不在邻接表中，添加
        if (!adjList.ContainsKey(v2))
            adjList[v2] = new List<int>();

        // 添加边（双向）
        adjList[v1].Add(v2);
        adjList[v2].Add(v1);
    }

    // ========== 广度优先搜索（BFS） ==========

    /// <summary>
    /// 广度优先搜索：按层次遍历图
    /// 使用队列实现
    /// </summary>
    public void BFS(int start)
    {
        // 记录已访问的顶点
        HashSet<int> visited = new HashSet<int>();
        
        // 使用队列存储待访问的顶点
        Queue<int> queue = new Queue<int>();

        // 从起始顶点开始
        visited.Add(start);
        queue.Enqueue(start);

        // 当队列不为空时继续
        while (queue.Count > 0)
        {
            // 出队一个顶点
            int v = queue.Dequeue();
            
            // 访问当前顶点
            Console.Write(v + " ");

            // 遍历所有邻居
            if (adjList.ContainsKey(v))
            {
                foreach (int neighbor in adjList[v])
                {
                    // 如果邻居未被访问
                    if (!visited.Contains(neighbor))
                    {
                        visited.Add(neighbor);
                        queue.Enqueue(neighbor);
                    }
                }
            }
        }
    }

    // ========== 深度优先搜索（DFS） ==========

    /// <summary>
    /// 深度优先搜索：尽可能深地遍历图
    /// 使用递归实现
    /// </summary>
    public void DFS(int start)
    {
        HashSet<int> visited = new HashSet<int>();
        DFSRecursive(start, visited);
    }

    /// <summary>
    /// DFS 递归辅助方法
    /// </summary>
    private void DFSRecursive(int v, HashSet<int> visited)
    {
        // 标记当前顶点为已访问
        visited.Add(v);
        
        // 访问当前顶点
        Console.Write(v + " ");

        // 遍历所有邻居
        if (adjList.ContainsKey(v))
        {
            foreach (int neighbor in adjList[v])
            {
                // 如果邻居未被访问，递归访问
                if (!visited.Contains(neighbor))
                {
                    DFSRecursive(neighbor, visited);
                }
            }
        }
    }
}
```

### 二叉堆

以下是一个完整的**泛型二叉堆**实现，默认作为**最小堆**，并通过 `IComparer<T>` 支持自定义排序（可轻松切换为最大堆）。代码包含详细注释，涵盖核心原理、操作流程与复杂度。

```c#
using System;
using System.Collections;
using System.Collections.Generic;

/// <summary>
/// 基于数组的泛型二叉堆（默认最小堆）。
/// 通过传入自定义比较器可变为最大堆或其他顺序。
/// </summary>
/// <typeparam name="T">堆中元素的类型</typeparam>
public class BinaryHeap<T> : IEnumerable<T>
{
    // 默认初始容量
    private const int DefaultCapacity = 4;

    // 元素存储数组，索引从0开始使用
    private T[] _items;
    // 当前元素个数
    private int _size;
    // 用于确定元素之间顺序的比较器
    private readonly IComparer<T> _comparer;

    /// <summary>
    /// 获取堆中当前元素的数量。
    /// </summary>
    public int Count => _size;

    /// <summary>
    /// 获取堆是否为空。
    /// </summary>
    public bool IsEmpty => _size == 0;

    /// <summary>
    /// 使用默认比较器初始化一个空堆（最小堆，要求 T 实现 IComparable）。
    /// </summary>
    public BinaryHeap() : this(DefaultCapacity, Comparer<T>.Default)
    {
    }

    /// <summary>
    /// 使用指定的初始容量初始化一个空堆。
    /// </summary>
    /// <param name="capacity">初始容量</param>
    public BinaryHeap(int capacity) : this(capacity, Comparer<T>.Default)
    {
    }

    /// <summary>
    /// 使用自定义比较器初始化一个空堆。
    /// </summary>
    /// <param name="comparer">用于比较元素的比较器</param>
    public BinaryHeap(IComparer<T> comparer) : this(DefaultCapacity, comparer)
    {
    }

    /// <summary>
    /// 使用指定初始容量和比较器初始化一个空堆。
    /// </summary>
    /// <param name="capacity">初始容量</param>
    /// <param name="comparer">元素比较器</param>
    public BinaryHeap(int capacity, IComparer<T> comparer)
    {
        if (capacity < 0)
            throw new ArgumentOutOfRangeException(nameof(capacity), "容量不能为负数。");
        _items = new T[capacity];
        _comparer = comparer ?? throw new ArgumentNullException(nameof(comparer));
    }

    /// <summary>
    /// 从一个现有集合构建堆（Floyd 建堆法，时间复杂度 O(n)）。
    /// </summary>
    /// <param name="collection">要放入堆的集合</param>
    /// <param name="comparer">元素比较器</param>
    public BinaryHeap(IEnumerable<T> collection, IComparer<T> comparer = null)
    {
        _comparer = comparer ?? Comparer<T>.Default;
        if (collection == null)
            throw new ArgumentNullException(nameof(collection));

        // 将集合转换为数组，便于原地建堆
        var list = new List<T>(collection);
        _items = list.ToArray();
        _size = _items.Length;

        // 如果数组长度不够（集合可能为空），则分配默认容量
        if (_size == 0)
        {
            _items = new T[DefaultCapacity];
            return;
        }

        // 从最后一个非叶子节点开始，依次下沉，构建合法堆
        int lastNonLeaf = (_size / 2) - 1;
        for (int i = lastNonLeaf; i >= 0; i--)
        {
            HeapifyDown(i);
        }
    }

    /// <summary>
    /// 查看堆顶元素（不移除）。
    /// </summary>
    /// <returns>堆顶元素</returns>
    /// <exception cref="InvalidOperationException">堆为空时抛出</exception>
    public T Peek()
    {
        if (_size == 0)
            throw new InvalidOperationException("堆为空。");
        return _items[0];
    }

    /// <summary>
    /// 尝试查看堆顶元素。
    /// </summary>
    /// <param name="item">输出的堆顶元素，如果堆为空则为默认值</param>
    /// <returns>如果堆不为空返回 true，否则 false</returns>
    public bool TryPeek(out T item)
    {
        if (_size == 0)
        {
            item = default;
            return false;
        }
        item = _items[0];
        return true;
    }

    /// <summary>
    /// 将元素入堆，时间复杂度 O(log n)。
    /// </summary>
    /// <param name="item">要插入的元素</param>
    public void Push(T item)
    {
        // 确保容量足够
        if (_size == _items.Length)
        {
            Grow();
        }

        // 将新元素放在数组末尾
        _items[_size] = item;
        // 从新索引开始上浮，恢复堆序
        HeapifyUp(_size);
        _size++;
    }

    /// <summary>
    /// 移除并返回堆顶元素，时间复杂度 O(log n)。
    /// </summary>
    /// <returns>堆顶元素</returns>
    /// <exception cref="InvalidOperationException">堆为空时抛出</exception>
    public T Pop()
    {
        if (_size == 0)
            throw new InvalidOperationException("堆为空。");

        T root = _items[0];
        _size--;

        // 如果移除后还有元素，将最后一个元素移到堆顶，并下沉
        if (_size > 0)
        {
            _items[0] = _items[_size];
            HeapifyDown(0);
        }

        // 清除引用，帮助垃圾回收
        _items[_size] = default;
        return root;
    }

    /// <summary>
    /// 尝试移除堆顶元素。
    /// </summary>
    /// <param name="item">输出的堆顶元素</param>
    /// <returns>成功移除返回 true，堆为空返回 false</returns>
    public bool TryPop(out T item)
    {
        if (_size == 0)
        {
            item = default;
            return false;
        }
        item = Pop();
        return true;
    }

    /// <summary>
    /// 清空堆。
    /// </summary>
    public void Clear()
    {
        // 清除引用，帮助 GC
        for (int i = 0; i < _size; i++)
        {
            _items[i] = default;
        }
        _size = 0;
    }

    /// <summary>
    /// 返回堆中所有元素的无序副本数组。
    /// </summary>
    public T[] ToArray()
    {
        T[] result = new T[_size];
        Array.Copy(_items, result, _size);
        return result;
    }

    // ---------- 私有的堆操作 ----------

    /// <summary>
    /// 上浮操作：将索引处的元素不断与父节点比较交换，直到满足堆序。
    /// 对于最小堆，如果当前元素小于父节点则上浮。
    /// </summary>
    /// <param name="index">开始上浮的索引</param>
    private void HeapifyUp(int index)
    {
        // 当前元素所在位置
        int current = index;
        // 要上浮的元素值
        T item = _items[current];

        // 只要当前索引不是根（索引 0）
        while (current > 0)
        {
            int parentIndex = (current - 1) / 2;   // 父节点索引计算
            T parent = _items[parentIndex];

            // 如果当前元素与父节点已满足堆序（默认：父 <= 子），则停止
            if (_comparer.Compare(parent, item) <= 0)
                break;

            // 否则将父节点下移，并继续向上
            _items[current] = parent;
            current = parentIndex;
        }
        // 最终放置上浮元素
        _items[current] = item;
    }

    /// <summary>
    /// 下沉操作：将索引处的元素不断与较小的子节点比较交换，直到满足堆序。
    /// 对于最小堆，如果当前元素大于子节点中较小的那个则下沉。
    /// </summary>
    /// <param name="index">开始下沉的索引</param>
    private void HeapifyDown(int index)
    {
        int current = index;
        T item = _items[current];

        // 只有当 current 存在子节点时才继续
        while (true)
        {
            int leftChildIndex = 2 * current + 1;   // 左子索引
            if (leftChildIndex >= _size)
                break;  // 没有子节点，终止

            int rightChildIndex = 2 * current + 2;  // 右子索引
            // 找出较小的子节点索引（根据比较器）
            int smallerChildIndex = leftChildIndex;
            if (rightChildIndex < _size && _comparer.Compare(_items[rightChildIndex], _items[leftChildIndex]) < 0)
            {
                smallerChildIndex = rightChildIndex;
            }

            // 如果当前元素不大于较小孩子，则满足堆序，停止
            if (_comparer.Compare(item, _items[smallerChildIndex]) <= 0)
                break;

            // 否则将较小孩子上移，并继续下沉
            _items[current] = _items[smallerChildIndex];
            current = smallerChildIndex;
        }
        _items[current] = item;
    }

    /// <summary>
    /// 扩容内部数组，倍增策略。
    /// </summary>
    private void Grow()
    {
        int newCapacity = _items.Length == 0 ? DefaultCapacity : _items.Length * 2;
        // 防止容量溢出（当数组长度很大时）
        if (newCapacity < _items.Length && _items.Length > 0)
            newCapacity = int.MaxValue;
        Array.Resize(ref _items, newCapacity);
    }

    // ---------- 枚举支持 ----------
    /// <summary>
    /// 返回一个枚举器，遍历顺序不保证有序。
    /// </summary>
    public IEnumerator<T> GetEnumerator()
    {
        for (int i = 0; i < _size; i++)
        {
            yield return _items[i];
        }
    }

    IEnumerator IEnumerable.GetEnumerator() => GetEnumerator();
}
```

#### 注释要点概括

- **数组映射**：索引 `i` 的左子为 `2i+1`，右子为 `2i+2`，父为 `(i-1)/2`。

- **上浮（HeapifyUp）**：插入时将新元素放在末尾，然后沿树向上交换，直到比父节点大（最小堆）。采用“腾出位置，最后放置”的优化，减少交换次数。

- **下沉（HeapifyDown）**：删除堆顶后，用末尾元素填充根，然后与较小孩子交换，采用同样的优化。

- **建堆**：构造函数中从最后一个非叶子节点向前调用下沉，线性时间 `O(n)` 建堆。

- **比较器**：通过 `IComparer<T>` 抽象比较规则，类本身不假定顺序，使得同一份代码支持最小堆、最大堆或自定义优先级。

  

#### 这是基于 labuladong 教程中 `MyPriorityQueue` 的 C# 实现。

C# 中泛型数组不能直接像 Java 那样通过 `(T[]) new Object[n]` 强转（会抛出异常），通常使用 `new T[capacity]`（要求 `T` 有默认构造函数或为引用类型）或者底层用 `object[]` 存储并强制转换。为了保持代码简洁且符合 C# 习惯，这里使用 `List<T>` 作为底层存储的变体，或者更贴近原意地使用 `T[]` 并配合 `Activator` 或直接初始化。

考虑到教程强调的是**数组模拟二叉堆**的核心逻辑，下面使用 `T[]` 实现，并利用 C# 的 `Comparer<T>.Default` 来处理比较逻辑（相当于 Java 的 `Comparator`）。



```C#
using System;
using System.Collections.Generic;

/// <summary>
/// 基于数组实现的泛型优先级队列（小顶堆）
/// 对应 labuladong 教程中的 MyPriorityQueue
/// </summary>
/// <typeparam name="T">元素类型，必须实现 IComparable 接口或提供自定义比较器</typeparam>
public class MyPriorityQueue<T> where T : IComparable<T>
{
    // 底层使用数组实现二叉堆
    private T[] heap;
    
    // 堆中元素的数量
    private int size;
    
    // 比较器，用于决定元素的优先级顺序
    // C# 中通常使用 IComparer<T> 或让 T 实现 IComparable<T>
    private readonly IComparer<T> comparer;

    /// <summary>
    /// 构造函数
    /// </summary>
    /// <param name="capacity">初始容量</param>
    /// <param name="comparer">自定义比较器，如果为 null 则使用默认比较器</param>
    public MyPriorityQueue(int capacity, IComparer<T>? comparer = null)
    {
        // 初始化数组
        heap = new T[capacity];
        size = 0;
        // 如果未提供比较器，则使用默认比较器（要求 T 实现 IComparable<T>）
        this.comparer = comparer ?? Comparer<T>.Default;
    }

    /// <summary>
    /// 返回队列中的元素个数
    /// </summary>
    public int Size()
    {
        return size;
    }

    /// <summary>
    /// 判断队列是否为空
    /// </summary>
    public bool IsEmpty()
    {
        return size == 0;
    }

    // --- 辅助方法：计算索引 ---

    /// <summary>
    /// 获取父节点的索引
    /// 公式: (node - 1) / 2
    /// </summary>
    private int Parent(int node)
    {
        return (node - 1) / 2;
    }

    /// <summary>
    /// 获取左子节点的索引
    /// 公式: node * 2 + 1
    /// </summary>
    private int Left(int node)
    {
        return node * 2 + 1;
    }

    /// <summary>
    /// 获取右子节点的索引
    /// 公式: node * 2 + 2
    /// </summary>
    private int Right(int node)
    {
        return node * 2 + 2;
    }

    /// <summary>
    /// 交换数组中两个位置的元素
    /// </summary>
    private void Swap(int i, int j)
    {
        T temp = heap[i];
        heap[i] = heap[j];
        heap[j] = temp;
    }

    // --- 核心 API ---

    /// <summary>
    /// 查：返回堆顶元素（最小值），时间复杂度 O(1)
    /// </summary>
    public T Peek()
    {
        if (IsEmpty())
        {
            throw new InvalidOperationException("Priority queue underflow");
        }
        return heap[0];
    }

    /// <summary>
    /// 增：向堆中插入一个元素，时间复杂度 O(logN)
    /// 核心步骤：
    /// 1. 将新元素添加到数组末尾（保持完全二叉树结构）
    /// 2. 执行 swim 上浮操作，恢复堆性质
    /// </summary>
    public void Push(T x)
    {
        // 如果数组满了，进行扩容（加倍）
        if (size == heap.Length)
        {
            Resize(2 * heap.Length);
        }
        
        // 1. 把新元素追加到最后
        heap[size] = x;
        
        // 2. 然后上浮到正确位置
        Swim(size);
        
        // 元素数量加 1
        size++;
    }

    /// <summary>
    /// 删：删除并返回堆顶元素（最小值），时间复杂度 O(logN)
    /// 核心步骤：
    /// 1. 记录堆顶元素
    /// 2. 将堆底元素移到堆顶
    /// 3. 执行 sink 下沉操作，恢复堆性质
    /// 4. 必要时缩容
    /// </summary>
    public T Pop()
    {
        if (IsEmpty())
        {
            throw new InvalidOperationException("Priority queue underflow");
        }

        // 1. 记录堆顶元素（返回值）
        T res = heap[0];
        
        // 2. 把堆底元素放到堆顶
        // 注意：先交换，再减少 size，这样 heap[size-1] 就是原来的堆底元素
        Swap(0, size - 1);
        
        // 避免对象游离（帮助 GC 回收不再使用的对象引用）
        heap[size - 1] = default(T)!; 
        
        // 元素数量减 1
        size--;
        
        // 3. 然后下沉到正确位置，恢复堆性质
        Sink(0);
        
        // 4. 缩容：如果元素数量少于数组容量的 1/4，则减半数组容量
        // 防止频繁扩容缩容带来的震荡，通常设定下限为 1/4
        if (size > 0 && size == heap.Length / 4)
        {
            Resize(heap.Length / 2);
        }
        
        return res;
    }

    // --- 堆性质维护核心算法 ---

    /// <summary>
    /// 上浮操作：将节点 node 向上移动，直到它不小于其父节点
    /// 时间复杂度：O(logN)，即树的高度
    /// 
    /// 小顶堆逻辑：如果当前节点比父节点小，就交换，继续向上比较
    /// </summary>
    private void Swim(int node)
    {
        // 只要没到根节点 (node > 0) 且 当前节点比父节点小
        while (node > 0 && comparer.Compare(heap[Parent(node)], heap[node]) > 0)
        {
            // 交换当前节点和父节点
            Swap(Parent(node), node);
            
            // 指针上移，继续检查新的位置
            node = Parent(node);
        }
    }

    /// <summary>
    /// 下沉操作：将节点 node 向下移动，直到它不大于其子节点
    /// 时间复杂度：O(logN)，即树的高度
    /// 
    /// 小顶堆逻辑：找出左右子节点中较小的那个，如果比当前节点小，就交换，继续向下比较
    /// </summary>
    private void Sink(int node)
    {
        // 只要左子节点存在（说明还有子节点），就继续循环
        // 注意：完全二叉树中，如果左子节点不存在，右子节点一定也不存在
        while (Left(node) < size)
        {
            // 假设左子节点是最小的
            int min = Left(node);
            
            // 如果右子节点也存在，且右子节点比左子节点更小，则更新 min 为右子节点
            if (Right(node) < size && comparer.Compare(heap[Right(node)], heap[min]) < 0)
            {
                min = Right(node);
            }
            
            // 如果当前节点已经比最小的子节点还小（或相等），说明满足堆性质，停止下沉
            if (comparer.Compare(heap[node], heap[min]) <= 0)
            {
                break;
            }
            
            // 否则，交换当前节点和最小的子节点
            Swap(node, min);
            
            // 指针下移，继续检查新的位置
            node = min;
        }
    }

    /// <summary>
    /// 调整堆底层数组的大小
    /// </summary>
    /// <param name="capacity">新的容量</param>
    private void Resize(int capacity)
    {
        // 创建新数组
        T[] temp = new T[capacity];
        
        // 复制原有元素
        for (int i = 0; i < size; i++)
        {
            temp[i] = heap[i];
        }
        
        // 替换旧数组
        heap = temp;
    }
}

// 测试代码
public class Program
{
    public static void Main(string[] args)
    {
        // 创建一个初始容量为 3 的小顶堆
        // 使用默认比较器（整数自然顺序）
        var pq = new MyPriorityQueue<int>(3);
        
        pq.Push(3);
        pq.Push(1);
        pq.Push(4);
        pq.Push(1);
        pq.Push(5);
        pq.Push(9);
        
        // 依次弹出，应为有序：1, 1, 3, 4, 5, 9
        while (!pq.IsEmpty())
        {
            Console.WriteLine(pq.Pop());
        }
    }
}
```

#### 关键点解析

1. **泛型约束 `where T : IComparable<T>`**： Java 版本使用了 `Comparator<? super T>`，C# 中对应的做法是让类型 `T` 实现 `IComparable<T>` 接口，或者在构造函数中传入 `IComparer<T>`。代码中使用了 `Comparer<T>.Default` 作为默认比较器，这样可以支持 `int`, `string` 等内置类型以及任何实现了比较接口的自定义类型。
2. **数组索引计算**： 与教程一致，使用从 0 开始的数组索引：
   - 父节点：`(node - 1) / 2`
   - 左子节点：`node * 2 + 1`
   - 右子节点：`node * 2 + 2`
3. **Swim（上浮）**： 用于 `Push` 操作。新元素加到末尾后，可能比父节点小（违反小顶堆规则），所以不断与父节点交换，直到到达根节点或不再小于父节点。
4. **Sink（下沉）**： 用于 `Pop` 操作。堆顶元素被移除后，将最后一个元素移到堆顶，它可能比子节点大（违反小顶堆规则），所以不断与较小的子节点交换，直到到达叶子节点或不再大于子节点。
5. **动态扩容与缩容**：
   - `Push` 时如果 `size == heap.Length`，调用 `Resize(2 * heap.Length)` 扩容。
   - `Pop` 后如果 `size == heap.Length / 4`，调用 `Resize(heap.Length / 2)` 缩容，节省内存。
6. **避免对象游离**： 在 `Pop` 中，`heap[size - 1] = default(T)!;` 这一步很重要。因为数组仍然持有该对象的引用，如果不置空，垃圾回收器（GC）无法回收该对象，可能导致内存泄漏。

### 3.1 排序算法

#### 核心概念解析

**什么是排序？**

- 排序是将一组数据按某种顺序（升序/降序）排列的过程
- 是计算机科学中最基本、最常见的算法问题

**排序算法的分类**

```
排序算法分类：
├── 比较排序（Comparison Sort）
│   ├── O(n²)：冒泡排序、选择排序、插入排序
│   ├── O(n log n)：快速排序、归并排序、堆排序
│   └── O(n log² n)：希尔排序
│
└── 非比较排序（Non-comparison Sort）
    ├── 计数排序
    ├── 桶排序
    └── 基数排序
```

**时间复杂度详解**

| 复杂度        | 名称   | 含义        | 举例   |
| ---------- | ---- | --------- | ---- |
| O(1)       | 常数时间 | 执行时间与输入无关 | 数组访问 |
| O(log n)   | 对数时间 | 每步减半      | 二分搜索 |
| O(n)       | 线性时间 | 与输入成正比    | 遍历数组 |
| O(n log n) | 线性对数 | 快速排序平均复杂度 | 高效排序 |
| O(n²)      | 平方时间 | 嵌套循环      | 冒泡排序 |

**稳定性**

- **稳定排序**：排序后相同元素的相对位置不变
- **不稳定排序**：排序后相同元素的相对位置可能改变

```
稳定性示例：
原数组: [(3, A), (1, B), (3, C)]  // 3 为值，A/C 为标记
稳定排序: [(1, B), (3, A), (3, C)]  ← 3的相对位置不变
不稳定:   [(1, B), (3, C), (3, A)]  ← 3的相对位置改变了
```

#### 冒泡排序

```csharp
/// <summary>
/// 冒泡排序：通过相邻元素比较交换，将最大元素逐步"冒泡"到末尾
/// 时间复杂度：O(n²)
/// 空间复杂度：O(1)（原地排序）
/// 稳定性：稳定
/// </summary>
public void BubbleSort(int[] arr)
{
    int n = arr.Length;
    
    // 外层循环：控制排序轮数，共 n-1 轮
    for (int i = 0; i < n - 1; i++)
    {
        // 内层循环：每轮比较到未排序部分的倒数第二个元素
        // 每轮结束后，最大的 i+1 个元素已排好序
        for (int j = 0; j < n - i - 1; j++)
        {
            // 如果前一个元素大于后一个，交换它们
            if (arr[j] > arr[j + 1])
            {
                // C# 7.0+ 的元组交换语法
                (arr[j], arr[j + 1]) = (arr[j + 1], arr[j]);
            }
        }
    }
}
// 时间复杂度: O(n²) - 两层嵌套循环
// 空间复杂度: O(1) - 原地排序，无需额外空间
// 稳定性: 稳定排序（相等元素相对位置不变）
```

#### 快速排序

```csharp
/// <summary>
/// 快速排序：分治思想，选择基准元素，将数组分成两部分
/// 时间复杂度：平均 O(n log n)，最坏 O(n²)
/// 空间复杂度：O(log n)（递归栈深度）
/// 稳定性：不稳定
/// </summary>
public void QuickSort(int[] arr, int low, int high)
{
    // 递归终止条件：区间无效或只有一个元素
    if (low < high)
    {
        // 分区操作：返回基准元素的最终位置
        int pi = Partition(arr, low, high);
        
        // 递归排序左半部分 [low, pi-1]
        QuickSort(arr, low, pi - 1);
        
        // 递归排序右半部分 [pi+1, high]
        QuickSort(arr, pi + 1, high);
    }
}

/// <summary>
/// 分区函数：选择最后一个元素作为基准，将数组分为两部分
/// 小于等于基准的在左边，大于基准的在右边
/// </summary>
/// <returns>基准元素的最终位置</returns>
private int Partition(int[] arr, int low, int high)
{
    // 选择最后一个元素作为基准
    int pivot = arr[high];
    
    // i 表示小于等于基准的区域的右边界
    int i = low - 1;

    // 遍历除基准外的所有元素
    for (int j = low; j < high; j++)
    {
        // 如果当前元素小于等于基准
        if (arr[j] <= pivot)
        {
            i++;  // 扩大小于等于区域
            // 交换到小于等于区域
            (arr[i], arr[j]) = (arr[j], arr[i]);
        }
    }
    
    // 将基准元素放到正确位置（i+1）
    (arr[i + 1], arr[high]) = (arr[high], arr[i + 1]);
    
    return i + 1;  // 返回基准位置
}
// 时间复杂度: 平均 O(n log n), 最坏 O(n²)（已排序数组且选择端点为基准）
// 空间复杂度: O(log n)（递归调用栈深度）
// 稳定性: 不稳定排序
```

#### 归并排序

```csharp
/// <summary>
/// 归并排序：分治思想，先拆分再合并
/// 时间复杂度：O(n log n)
/// 空间复杂度：O(n)（需要额外数组存储）
/// 稳定性：稳定
/// </summary>
public void MergeSort(int[] arr, int left, int right)
{
    // 递归终止条件：区间无效或只有一个元素
    if (left < right)
    {
        // 计算中点，避免溢出
        int mid = left + (right - left) / 2;
        
        // 递归排序左半部分 [left, mid]
        MergeSort(arr, left, mid);
        
        // 递归排序右半部分 [mid+1, right]
        MergeSort(arr, mid + 1, right);
        
        // 合并两个有序部分
        Merge(arr, left, mid, right);
    }
}

/// <summary>
/// 合并函数：将两个有序子数组合并为一个有序数组
/// </summary>
private void Merge(int[] arr, int left, int mid, int right)
{
    // 计算两个子数组的长度
    int n1 = mid - left + 1;  // 左子数组长度
    int n2 = right - mid;     // 右子数组长度

    // 创建临时数组存储两个子数组
    int[] L = new int[n1];
    int[] R = new int[n2];

    // 将数据复制到临时数组
    for (int x = 0; x < n1; x++)
        L[x] = arr[left + x];
    for (int x = 0; x < n2; x++)
        R[x] = arr[mid + 1 + x];

    // 合并两个有序数组
    int i = 0;  // 左子数组指针
    int j = 0;  // 右子数组指针
    int k = left;  // 原数组指针

    // 比较并合并
    while (i < n1 && j < n2)
    {
        if (L[i] <= R[j])
            arr[k++] = L[i++];
        else
            arr[k++] = R[j++];
    }

    // 复制左子数组剩余元素
    while (i < n1)
        arr[k++] = L[i++];
    
    // 复制右子数组剩余元素
    while (j < n2)
        arr[k++] = R[j++];
}
// 时间复杂度: O(n log n) - 拆分 log n 层，每层合并 O(n)
// 空间复杂度: O(n) - 需要额外的临时数组
// 稳定性: 稳定排序
```

### 3.2 二分搜索

#### 核心概念解析

**什么是二分搜索？**

- 二分搜索是一种在**有序数组**中查找目标元素的高效算法
- 核心思想：每次将搜索范围缩小一半，复杂度 **O(log n)**

```
二分搜索过程示例：
数组: [1, 3, 5, 7, 9, 11, 13, 15]  目标: 11

第1次: left=0, right=7, mid=3, arr[3]=7 < 11 → right=mid-1=2
       搜索范围: [5, 7, 9, 11, 13, 15]  (减半!)

第2次: left=4, right=7, mid=5, arr[5]=11 == 11 → 找到!
       搜索范围: [11, 13, 15]

查找次数: 2次  (log₂8 = 3，最多3次)
```

**二分搜索的适用条件**

1. **有序数组**（升序或降序）
2. **能够随机访问**（数组比链表好）
3. **数据量较大**时优势明显（10亿数据只需30次比较）

**二分搜索的难点**

1. **边界条件**：`left <= right` 还是 `left < right`？
2. **更新方式**：`mid ± 1` 还是 `mid`？
3. **防止溢出**：`left + (right - left) / 2` 而非 `(left + right) / 2`

**二分搜索的变体**

1. **查找第一个等于目标的位置**
2. **查找最后一个等于目标的位置**
3. **查找第一个大于等于目标的位置**
4. **查找最后一个小于等于目标的位置**
5. **旋转数组中的搜索**（LeetCode 33）

**为什么用** **`left + (right - left) / 2`？**

- 当 left 和 right 很大时，`left + right` 可能**溢出**（超过 int.MaxValue）
- `right - left` 不会超过 int 范围
- 例如：`left = 2^30, right = 2^30`，用 `left + right` 会变负数

#### 二分搜索模板

```csharp
/// <summary>
/// 标准二分搜索：在有序数组中查找目标值的位置
/// 时间复杂度：O(log n)
/// 空间复杂度：O(1)
/// </summary>
/// <param name="arr">有序数组（升序）</param>
/// <param name="target">目标值</param>
/// <returns>目标值索引，未找到返回 -1</returns>
public int BinarySearch(int[] arr, int target)
{
    int left = 0;                    // 左边界
    int right = arr.Length - 1;      // 右边界

    // 循环条件：搜索区间不为空
    while (left <= right)
    {
        // 计算中点，防止溢出
        int mid = left + (right - left) / 2;

        if (arr[mid] == target)
            return mid;              // 找到目标，返回索引
        else if (arr[mid] < target)
            left = mid + 1;          // 目标在右半部分
        else
            right = mid - 1;         // 目标在左半部分
    }
    return -1;  // 未找到目标
}

// ========== 二分搜索变体 ==========

/// <summary>
/// 找左边界：查找第一个大于等于目标值的位置
/// 适用于有序数组中存在重复元素的情况
/// </summary>
public int LeftBound(int[] arr, int target)
{
    int left = 0, right = arr.Length;
    
    // 左闭右开区间 [left, right)
    while (left < right)
    {
        int mid = left + (right - left) / 2;
        if (arr[mid] < target)
            left = mid + 1;  // 目标在右侧
        else
            right = mid;     // 目标在左侧（包含mid）
    }
    return left;  // 返回插入位置
}

/// <summary>
/// 找右边界：查找最后一个小于等于目标值的位置
/// </summary>
public int RightBound(int[] arr, int target)
{
    int left = 0, right = arr.Length;
    while (left < right)
    {
        int mid = left + (right - left) / 2;
        if (arr[mid] <= target)
            left = mid + 1;
        else
            right = mid;
    }
    return left - 1;
}
```

#### 二分搜索应用

```csharp
/// <summary>
/// 在排序数组中查找元素的第一个和最后一个位置
/// 时间复杂度：O(log n)
/// </summary>
public int[] SearchRange(int[] nums, int target)
{
    int left = FindLeft(nums, target);
    int right = FindRight(nums, target);
    
    // 验证结果是否有效
    if (left <= right && right < nums.Length && nums[left] == target)
        return new int[] { left, right };
    return new int[] { -1, -1 };
}

/// <summary>
/// 查找第一个等于目标的位置
/// </summary>
private int FindLeft(int[] nums, int target)
{
    int left = 0, right = nums.Length - 1;
    while (left <= right)
    {
        int mid = left + (right - left) / 2;
        if (nums[mid] >= target)
            right = mid - 1;  // 继续向左找
        else
            left = mid + 1;
    }
    return left;
}

/// <summary>
/// 查找最后一个等于目标的位置
/// </summary>
private int FindRight(int[] nums, int target)
{
    int left = 0, right = nums.Length - 1;
    while (left <= right)
    {
        int mid = left + (right - left) / 2;
        if (nums[mid] <= target)
            left = mid + 1;  // 继续向右找
        else
            right = mid - 1;
    }
    return right;
}
```

### 4.1 递归与分治

#### 递归基础

```csharp
/// <summary>
/// 阶乘计算（递归实现）
/// n! = n * (n-1) * (n-2) * ... * 1
/// </summary>
public int Factorial(int n)
{
    // 递归终止条件：0! = 1, 1! = 1
    if (n <= 1) return 1;
    
    // 递归调用：n! = n * (n-1)!
    return n * Factorial(n - 1);
}

/// <summary>
/// 斐波那契数列（递归实现）
/// F(n) = F(n-1) + F(n-2), F(0)=0, F(1)=1
/// </summary>
public int Fibonacci(int n)
{
    // 递归终止条件
    if (n <= 1) return n;
    
    // 递归调用
    return Fibonacci(n - 1) + Fibonacci(n - 2);
}

/// <summary>
/// 斐波那契数列（记忆化优化）
/// 避免重复计算，时间复杂度从 O(2^n) 降为 O(n)
/// </summary>
public int FibonacciMemo(int n, Dictionary<int, int> memo)
{
    if (n <= 1) return n;
    
    // 如果已经计算过，直接返回
    if (memo.ContainsKey(n)) return memo[n];

    // 计算并缓存结果
    memo[n] = FibonacciMemo(n - 1, memo) + FibonacciMemo(n - 2, memo);
    return memo[n];
}
```

#### 分治法示例

```csharp
/// <summary>
/// 最大子数组和（LeetCode 53）
/// 使用分治法求解，时间复杂度：O(n log n)
/// 分治法核心思想：将问题分解为子问题，递归求解后合并结果
/// </summary>
/// <param name="nums">输入数组</param>
/// <returns>最大子数组和</returns>
public int MaxSubArray(int[] nums)
{
    // 调用分治递归函数，从整个数组开始
    return MaxSubArrayDivide(nums, 0, nums.Length - 1);
}

/// <summary>
/// 分治递归函数：求解 [left, right] 区间内的最大子数组和
/// 
/// 分治法的三个步骤：
/// 1. 分解（Divide）：将数组分成左右两部分
/// 2. 解决（Conquer）：递归求解左右子问题
/// 3. 合并（Combine）：综合左右解和跨越中点的解
/// 
/// 最大子数组和可能出现在三个位置：
/// 1. 完全在左半部分 [left, mid]
/// 2. 完全在右半部分 [mid+1, right]
/// 3. 跨越中间位置，包含 mid 和 mid+1
/// </summary>
/// <param name="nums">输入数组</param>
/// <param name="left">区间左边界</param>
/// <param name="right">区间右边界</param>
/// <returns>区间内的最大子数组和</returns>
private int MaxSubArrayDivide(int[] nums, int left, int right)
{
    // 递归终止条件：区间只有一个元素时，最大子数组就是它本身
    if (left == right)
    {
        return nums[left];
    }

    // 计算中点，使用 left + (right - left) / 2 避免溢出
    int mid = left + (right - left) / 2;
    
    // ========== 分解与递归求解 ==========
    
    // 递归计算左半部分 [left, mid] 的最大子数组和
    int leftSum = MaxSubArrayDivide(nums, left, mid);
    
    // 递归计算右半部分 [mid+1, right] 的最大子数组和
    int rightSum = MaxSubArrayDivide(nums, mid + 1, right);
    
    // ========== 合并阶段 ==========
    
    // 计算跨越中间位置的最大子数组和
    int crossSum = MaxCrossingSum(nums, left, mid, right);

    // 返回三者中的最大值：左半解、右半解、跨越解
    return Math.Max(Math.Max(leftSum, rightSum), crossSum);
}

/// <summary>
/// 计算跨越中间位置的最大子数组和
/// 必须包含 mid 和 mid+1 两个元素
/// 
/// 计算方法：
/// 1. 从中点向左扫描，找到以 mid 结尾的最大和
/// 2. 从中点向右扫描，找到以 mid+1 开头的最大和
/// 3. 两部分相加即为跨越中点的最大和
/// </summary>
/// <param name="nums">输入数组</param>
/// <param name="left">区间左边界</param>
/// <param name="mid">区间中点</param>
/// <param name="right">区间右边界</param>
/// <returns>跨越中点的最大子数组和</returns>
private int MaxCrossingSum(int[] nums, int left, int mid, int right)
{
    // ========== 向左扫描 ==========
    int sum = 0;                // 当前累加和
    int leftSum = int.MinValue; // 向左扫描的最大和（初始为负无穷）
    
    // 从 mid 向左遍历到 left
    for (int i = mid; i >= left; i--)
    {
        sum += nums[i];  // 累加当前元素
        // 更新最大和（取当前最大值或新的累加和）
        leftSum = Math.Max(leftSum, sum);
    }

    // ========== 向右扫描 ==========
    sum = 0;                  // 重置累加和
    int rightSum = int.MinValue; // 向右扫描的最大和
    
    // 从 mid+1 向右遍历到 right
    for (int i = mid + 1; i <= right; i++)
    {
        sum += nums[i];      // 累加当前元素
        rightSum = Math.Max(rightSum, sum);
    }

    // 返回跨越中点的最大和：左半最大和 + 右半最大和
    return leftSum + rightSum;
}

// ========== 算法复杂度分析 ==========
// 时间复杂度：O(n log n)
//   - 分解：每次分成两半，共 log n 层递归
//   - 每层合并：每层处理 O(n) 个元素
//   - 总复杂度：O(n) * O(log n) = O(n log n)
//
// 空间复杂度：O(log n)
//   - 递归调用栈的深度为 O(log n)
//   - 未使用额外的数据结构
//
// 对比动态规划解法：
//   - DP 时间复杂度：O(n)
//   - DP 空间复杂度：O(1)（优化后）
//   - 分治法虽然复杂度稍高，但体现了分治思想的精髓
```

### 4.2 动态规划

#### 核心概念解析

**什么是动态规划？**

- 动态规划（Dynamic Programming，简称 DP）是一种**将复杂问题分解为简单子问题**的算法思想
- 核心：**记住已经计算过的结果**，避免重复计算

```
动态规划 vs 递归对比（以斐波那契为例）：

递归（暴力）：
Fib(5)
  ├─ Fib(4)
  │   ├─ Fib(3)
  │   │   ├─ Fib(2) → 1
  │   │   └─ Fib(1) → 1
  │   └─ Fib(2) → 1
  └─ Fib(3)
      ├─ Fib(2) → 1  ← 重复计算!
      └─ Fib(1) → 1

时间复杂度: O(2^n) - 指数爆炸!

动态规划：
Fib(0)=0, Fib(1)=1
Fib(2)=Fib(1)+Fib(0)=1
Fib(3)=Fib(2)+Fib(1)=2  ← 复用
Fib(4)=Fib(3)+Fib(2)=3
Fib(5)=Fib(4)+Fib(3)=5

时间复杂度: O(n)
```

**动态规划的三要素**

1. **最优子结构**：问题的最优解包含子问题的最优解
2. **无后效性**：一旦某个状态确定，后续决策不受影响
3. **状态转移方程**：大问题如何拆分为小问题

**动态规划的解题步骤**

```
1. 定义状态：dp[i] 表示什么？
2. 状态转移：dp[i] = ?
3. 初始化：dp[0], dp[1] 等
4. 遍历顺序：从前往后 or 从后往前
5. 推导结果：dp[n] 就是答案
```

**动态规划的两种实现方式**

| 方式        | 描述      | 空间复杂度 | 适用场景 |
| --------- | ------- | ----- | ---- |
| 自底向上（填表法） | 从小到大计算  | O(n)  | 所有情况 |
| 自顶向下（记忆化） | 递归 + 缓存 | O(n)  | 树形结构 |

**常见的动态规划问题**

1. **基础 DP**：斐波那契、爬楼梯、错排
2. **背包问题**：0-1 背包、完全背包
3. **子序列问题**：最长上升子序列、最长公共子序列
4. **字符串问题**：编辑距离、最长回文子串
5. **股票问题**：买卖股票最佳时机系列

```csharp
/// <summary>
/// 斐波那契数列（DP 空间优化）
/// 空间复杂度：O(1)，时间复杂度：O(n)
/// </summary>
public int FibonacciDP(int n)
{
    if (n <= 1) return n;
    
    // 使用两个变量保存前两项，避免使用数组
    int prev2 = 0, prev1 = 1;
    
    for (int i = 2; i <= n; i++)
    {
        int current = prev1 + prev2;
        prev2 = prev1;
        prev1 = current;
    }
    return prev1;
}

/// <summary>
/// 爬楼梯（LeetCode 70）
/// 每次可以爬 1 或 2 个台阶，求到达第 n 阶的方法数
/// </summary>
public int ClimbStairs(int n)
{
    if (n <= 2) return n;
    
    // dp[i] 表示到达第 i 阶的方法数
    int[] dp = new int[n + 1];
    dp[1] = 1;  // 第1阶：1种方法
    dp[2] = 2;  // 第2阶：2种方法（1+1 或 2）
    
    for (int i = 3; i <= n; i++)
    {
        // 状态转移：到达第i阶 = 从i-1阶爬1步 + 从i-2阶爬2步
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
}

/// <summary>
/// 爬楼梯（空间优化版）
/// 空间复杂度：O(1)
/// </summary>
public int ClimbStairsOptimized(int n)
{
    if (n <= 2) return n;
    
    int prev2 = 1, prev1 = 2;
    for (int i = 3; i <= n; i++)
    {
        int current = prev1 + prev2;
        prev2 = prev1;
        prev1 = current;
    }
    return prev1;
}
```

```csharp
/// <summary>
/// 分割等和子集（背包问题变体）
/// 判断数组是否可以分割成两个和相等的子集
/// </summary>
public bool CanPartition(int[] nums)
{
    int sum = nums.Sum();
    // 如果总和是奇数，无法分割
    if (sum % 2 != 0) return false;
    
    int target = sum / 2;
    
    // dp 存储可以组成的和
    HashSet<int> dp = new HashSet<int>();
    dp.Add(0);  // 初始状态：可以组成和为0
    
    foreach (int num in nums)
    {
        HashSet<int> newDp = new HashSet<int>();
        foreach (int x in dp)
        {
            newDp.Add(x);        // 不选当前数
            newDp.Add(x + num);  // 选当前数
        }
        dp = newDp;
    }
    return dp.Contains(target);
}

/// <summary>
/// 最长公共子序列（LCS）
/// 找出两个字符串的最长公共子序列长度
/// </summary>
public int LongestCommonSubsequence(string text1, string text2)
{
    int m = text1.Length, n = text2.Length;
    
    // dp[i,j] 表示 text1[0..i-1] 和 text2[0..j-1] 的 LCS 长度
    int[,] dp = new int[m + 1, n + 1];

    for (int i = 1; i <= m; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            if (text1[i - 1] == text2[j - 1])
                // 字符相等，LCS 长度加1
                dp[i, j] = dp[i - 1, j - 1] + 1;
            else
                // 字符不相等，取两种情况的最大值
                dp[i, j] = Math.Max(dp[i - 1, j], dp[i, j - 1]);
        }
    }
    return dp[m, n];
}

/// <summary>
/// 编辑距离（Levenshtein Distance）
/// 将 word1 转换成 word2 所需的最少操作数
/// 操作：插入、删除、替换
/// </summary>
public int MinDistance(string word1, string word2)
{
    int m = word1.Length, n = word2.Length;
    
    // dp[i,j] 表示 word1[0..i-1] 转换成 word2[0..j-1] 的最少操作数
    int[,] dp = new int[m + 1, n + 1];

    // 初始化：空字符串转换
    for (int i = 0; i <= m; i++) dp[i, 0] = i;  // 删除 i 次
    for (int j = 0; j <= n; j++) dp[0, j] = j;  // 插入 j 次

    for (int i = 1; i <= m; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            if (word1[i - 1] == word2[j - 1])
                // 字符相等，无需操作
                dp[i, j] = dp[i - 1, j - 1];
            else
                // 取三种操作的最小值 + 1
                dp[i, j] = Math.Min(dp[i - 1, j - 1],  // 替换
                                   Math.Min(dp[i - 1, j],  // 删除
                                            dp[i, j - 1])) + 1;  // 插入
        }
    }
    return dp[m, n];
}
```

### 4.3 贪心算法

#### 贪心基础

```csharp
/// <summary>
/// 分发糖果（LeetCode 135）
/// 每个孩子至少一个糖果，评分高的孩子必须比相邻孩子多
/// 贪心策略：两次遍历，分别处理左右约束
/// </summary>
public int Candy(int[] ratings)
{
    int n = ratings.Length;
    int[] candies = new int[n];
    candies[0] = 1;  // 第一个孩子至少1颗糖

    // 第一次遍历：从左到右，处理右边比左边评分高的情况
    for (int i = 1; i < n; i++)
    {
        if (ratings[i] > ratings[i - 1])
            candies[i] = candies[i - 1] + 1;  // 比左边多一颗
        else
            candies[i] = 1;  // 至少1颗
    }

    // 第二次遍历：从右到左，处理左边比右边评分高的情况
    for (int i = n - 2; i >= 0; i--)
    {
        if (ratings[i] > ratings[i + 1])
            candies[i] = Math.Max(candies[i], candies[i + 1] + 1);
    }

    return candies.Sum();
}

/// <summary>
/// 跳跃游戏（LeetCode 55）
/// 判断是否可以从第一个位置跳到最后一个位置
/// 贪心策略：维护最远可达位置
/// </summary>
public bool CanJump(int[] nums)
{
    int maxReach = 0;  // 当前最远可达位置
    
    for (int i = 0; i < nums.Length; i++)
    {
        // 如果当前位置超过最远可达位置，无法到达
        if (i > maxReach) return false;
        
        // 更新最远可达位置
        maxReach = Math.Max(maxReach, i + nums[i]);
        
        // 如果已经可以到达终点
        if (maxReach >= nums.Length - 1) return true;
    }
    return true;
}

/// <summary>
/// 活动选择问题
/// 在不重叠的情况下，选择最多的活动
/// 贪心策略：每次选择结束最早的活动
/// </summary>
public int MaxActivities(int[] start, int[] end)
{
    int n = start.Length;
    int[] idx = new int[n];
    for (int i = 0; i < n; i++) idx[i] = i;

    // 按结束时间排序
    Array.Sort(idx, (a, b) => end[a].CompareTo(end[b]));

    int count = 1;  // 至少选择第一个活动
    int lastEnd = end[idx[0]];

    for (int i = 1; i < n; i++)
    {
        // 如果当前活动开始时间 >= 上一个活动结束时间
        if (start[idx[i]] >= lastEnd)
        {
            count++;
            lastEnd = end[idx[i]];
        }
    }
    return count;
}
```

## 第五章：游戏开发常用算法

### 5.1 A\*寻路算法

#### 核心概念解析

*什么是 A* 算法？\*

- A\*（A-Star）是一种**启发式搜索**算法，用于在图中找到从起点到终点的**最优路径**
- 广泛应用于游戏中的**NPC 寻路**、**自动导航**等

*A* 算法的核心公式\*

```
F = G + H

其中：
- G：从起点到当前点的实际成本（已走距离）
- H：从当前点到终点的估计成本（启发函数）
- F：总成本（评估值），越小越先被选中
```

```
A*算法搜索过程示例：

S=起点  E=终点  #=障碍  .=可走

起始地图：
S . . #
. # . .
. . . E

第1步：检查 S 的邻居
  邻居(0,1): G=1, H=3, F=4
  邻居(1,0): G=1, H=2, F=3  ← 最优，先处理

第2步：处理 (1,0)
  邻居(1,1): G=2, H=2, F=4
  邻居(2,0): G=2, H=1, F=3  ← 最优

...依此类推直到找到 E
```

**启发函数 H 的选择**

1. **曼哈顿距离**（Manhattan）：`|x1-x2| + |y1-y2|`
   - 适合只能上下左右移动的场景（4方向）
2. **欧几里得距离**（Euclidean）：`√((x1-x2)² + (y1-y2)²)`
   - 适合可以任意方向移动的场景
3. **对角线距离**（Diagonal）：`max(|x1-x2|, |y1-y2|)`
   - 适合可以8方向移动的场景

*A* 算法的特点\*

- **优点**：找到最优路径，搜索效率高
- **缺点**：需要存储大量节点，内存开销大
- **注意**：启发函数 H 不能大于实际距离，否则可能找不到最优解

*A* vs 其他寻路算法\*

| 算法       | 特点           | 适用场景     |
| -------- | ------------ | -------- |
| BFS      | 找到最短路径，但遍历点多 | 小地图、迷宫   |
| Dijkstra | 考虑权重，但慢      | 有不同移动代价  |
| A\*      | 综合最优，效率高     | 游戏寻路（常用） |
| 贪心       | 速度快，但不一定最优   | 实时性要求高   |

**游戏开发中的优化**

1. **分层寻路**：大地图分成多个区域，先找区域路径
2. **路点系统**：预先计算关键点形成路径网络
3. **对象池**：复用搜索中的节点对象
4. **多线程**：在后台进行寻路计算

#### A\* 算法实现

```csharp
/// <summary>
/// A*寻路算法实现
/// 使用曼哈顿距离作为启发函数，支持4方向移动
/// </summary>
public class AStar
{
    /// <summary>
    /// 搜索节点
    /// </summary>
    private class Node
    {
        public int x, y;           // 坐标
        public int g;              // G值：起点到当前点的实际成本
        public int h;              // H值：当前点到终点的估计成本（启发函数）
        public int f => g + h;     // F值：总成本（G + H）
        public Node parent;        // 父节点（用于回溯路径）
    }

    private int[,] grid;    // 地图网格（0=可走，1=障碍）
    private int width;      // 地图宽度
    private int height;     // 地图高度

    /// <summary>
    /// 构造函数
    /// </summary>
    public AStar(int[,] grid)
    {
        this.grid = grid;
        this.width = grid.GetLength(0);
        this.height = grid.GetLength(1);
    }

    /// <summary>
    /// 查找从起点到终点的路径
    /// </summary>
    public List<Node> FindPath(int startX, int startY, int endX, int endY)
    {
        // 创建起点和终点节点
        Node startNode = new Node { x = startX, y = startY, g = 0, h = Heuristic(startX, startY, endX, endY) };
        Node endNode = new Node { x = endX, y = endY };

        // 开放列表：待处理的节点
        List<Node> openSet = new List<Node>();
        // 封闭列表：已处理的节点
        HashSet<Node> closedSet = new HashSet<Node>();

        openSet.Add(startNode);

        while (openSet.Count > 0)
        {
            // 找到F值最小的节点
            Node current = openSet[0];
            for (int i = 1; i < openSet.Count; i++)
            {
                if (openSet[i].f < current.f)
                    current = openSet[i];
            }

            // 从开放列表移除，加入封闭列表
            openSet.Remove(current);
            closedSet.Add(current);

            // 检查是否到达终点
            if (current.x == endNode.x && current.y == endNode.y)
            {
                return ReconstructPath(current);
            }

            // 遍历所有邻居
            foreach (Node neighbor in GetNeighbors(current))
            {
                // 跳过已处理的节点
                if (closedSet.Contains(neighbor)) continue;
                // 跳过障碍物
                if (grid[neighbor.x, neighbor.y] == 1) continue;

                // 计算从起点到邻居的临时G值
                int tentativeG = current.g + 1;

                // 检查邻居是否在开放列表中
                bool inOpen = openSet.Exists(n => n.x == neighbor.x && n.y == neighbor.y);
                
                // 如果不在开放列表中，或者找到了更优的路径
                if (!inOpen || tentativeG < neighbor.g)
                {
                    neighbor.g = tentativeG;
                    neighbor.h = Heuristic(neighbor.x, neighbor.y, endX, endY);
                    neighbor.parent = current;

                    if (!inOpen)
                        openSet.Add(neighbor);
                }
            }
        }

        return null;  // 无法到达终点
    }

    /// <summary>
    /// 启发函数：曼哈顿距离
    /// </summary>
    private int Heuristic(int x1, int y1, int x2, int y2)
    {
        return Math.Abs(x1 - x2) + Math.Abs(y1 - y2);
    }

    /// <summary>
    /// 获取相邻节点（4方向）
    /// </summary>
    private List<Node> GetNeighbors(Node node)
    {
        List<Node> neighbors = new List<Node>();
        // 上、右、下、左四个方向
        int[,] dirs = { { 0, 1 }, { 1, 0 }, { 0, -1 }, { -1, 0 } };

        for (int i = 0; i < 4; i++)
        {
            int nx = node.x + dirs[i, 0];
            int ny = node.y + dirs[i, 1];

            if (nx >= 0 && nx < width && ny >= 0 && ny < height)
            {
                neighbors.Add(new Node { x = nx, y = ny });
            }
        }
        return neighbors;
    }

    private List<Node> ReconstructPath(Node node)
    {
        List<Node> path = new List<Node>();
        while (node != null)
        {
            path.Add(node);
            node = node.parent;
        }
        path.Reverse();
        return path;
    }
}
```

### 5.2 对象池算法

#### 通用对象池

```csharp
/// <summary>
/// 泛型对象池实现
/// 原理：预先创建并缓存对象，需要时从池中获取，用完后放回
/// 目的：减少频繁创建和销毁对象带来的性能开销
/// </summary>
/// <typeparam name="T">池化对象类型（必须是引用类型且有默认构造函数）</typeparam>
public class ObjectPool<T> where T : class, new()
{
    // 存储空闲对象的栈（后进先出）
    private readonly Stack<T> pool;
    // 创建新对象的方法
    private readonly Func<T> createFunc;
    // 对象回收时的重置方法
    private readonly Action<T> resetAction;
    // 池的最大容量
    private readonly int maxSize;

    // 构造函数
    public ObjectPool(Func<T> createFunc, Action<T> resetAction = null, int maxSize = 100)
    {
        this.pool = new Stack<T>();
        this.createFunc = createFunc;
        this.resetAction = resetAction;
        this.maxSize = maxSize;
    }

    /// <summary>
    /// 从池中获取对象
    /// </summary>
    public T Get()
    {
        // 如果池中有空闲对象，弹出一个；否则创建新对象
        return pool.Count > 0 ? pool.Pop() : createFunc();
    }

    /// <summary>
    /// 将对象放回池中
    /// </summary>
    public void Return(T obj)
    {
        // 空对象不处理
        if (obj == null) return;
        
        // 如果有重置方法，先重置对象状态
        resetAction?.Invoke(obj);
        
        // 如果池未满，放回池中
        if (pool.Count < maxSize)
            pool.Push(obj);
        // 如果池已满，对象会被GC回收
    }

    // 获取池中对象数量
    public int Count => pool.Count;

    // 清空池
    public void Clear()
    {
        pool.Clear();
    }
}

// ========== 游戏开发中具体应用：子弹池 ==========

/// <summary>
/// 子弹类
/// </summary>
public class Bullet
{
    public Vector3 position;  // 位置
    public Vector3 velocity;  // 速度
    public float lifetime;    // 生命周期

    /// <summary>
    /// 重置子弹状态
    /// </summary>
    public void Reset()
    {
        velocity = Vector3.zero;
        lifetime = 0;
    }
}

/// <summary>
/// 子弹对象池
/// </summary>
public class BulletPool
{
    private ObjectPool<Bullet> pool;
    private List<Bullet> activeBullets = new List<Bullet>();  // 活跃的子弹列表

    public BulletPool()
    {
        // 创建子弹池
        pool = new ObjectPool<Bullet>(
            createFunc: () => new Bullet(),       // 如何创建新子弹
            resetAction: b => b.Reset(),          // 回收时如何重置
            maxSize: 100                          // 最大容量
        );
    }

    /// <summary>
    /// 生成子弹
    /// </summary>
    public Bullet Spawn(Vector3 pos, Vector3 vel)
    {
        // 从池中获取子弹
        Bullet bullet = pool.Get();
        
        // 设置初始状态
        bullet.position = pos;
        bullet.velocity = vel;
        bullet.lifetime = 3f;  // 3秒生命周期
        
        // 添加到活跃列表
        activeBullets.Add(bullet);
        
        return bullet;
    }

    /// <summary>
    /// 更新所有活跃子弹
    /// </summary>
    public void Update(float deltaTime)
    {
        for (int i = activeBullets.Count - 1; i >= 0; i--)
        {
            Bullet b = activeBullets[i];
            b.position += b.velocity * deltaTime;
            b.lifetime -= deltaTime;

            if (b.lifetime <= 0)
            {
                pool.Return(b);
                activeBullets.RemoveAt(i);
            }
        }
    }
}
```
