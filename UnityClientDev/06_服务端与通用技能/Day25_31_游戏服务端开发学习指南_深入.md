### 服务管理脚本

```bash
#!/bin/bash
# server_ctl.sh — 游戏服务器控制脚本

APP_NAME="GameServer"
BIN_PATH="./bin/$APP_NAME"
LOG_DIR="./logs"
PID_FILE="./run/$APP_NAME.pid"
CONFIG="./config/server.json"

# 确保目录存在
mkdir -p "$LOG_DIR" "./run"

start() {
    if [ -f "$PID_FILE" ]; then
        PID=$(cat "$PID_FILE")
        if kill -0 "$PID" 2>/dev/null; then
            echo "服务器已在运行 (PID: $PID)"
            exit 1
        fi
        rm -f "$PID_FILE"
    fi

    echo "启动 $APP_NAME ..."
    nohup "$BIN_PATH" --config "$CONFIG" > "$LOG_DIR/output.log" 2>&1 &
    PID=$!
    echo $PID > "$PID_FILE"
    echo "启动完成 (PID: $PID)"
}

stop() {
    if [ ! -f "$PID_FILE" ]; then
        echo "未找到 PID 文件"
        return
    fi

    PID=$(cat "$PID_FILE")
    echo "发送 SIGTERM 到进程 $PID ..."
    kill -15 "$PID"

    # 等待优雅退出（最多10秒）
    for i in $(seq 1 10); do
        if ! kill -0 "$PID" 2>/dev/null; then
            echo "进程已退出"
            rm -f "$PID_FILE"
            return 0
        fi
        sleep 1
    done

    # 超时强制杀死
    echo "超时，发送 SIGKILL ..."
    kill -9 "$PID" 2>/dev/null
    rm -f "$PID_FILE"
}

reload() {
    # 热重载配置（需要服务器支持 SIGHUP）
    if [ -f "$PID_FILE" ]; then
        PID=$(cat "$PID_FILE")
        kill -HUP "$PID"
        echo "已发送重载信号到 $PID"
    fi
}

status() {
    if [ -f "$PID_FILE" ]; then
        PID=$(cat "$PID_FILE")
        if kill -0 "$PID" 2>/dev/null; then
            echo "$APP_NAME 运行中 (PID: $PID)"
            # 显示资源占用
            ps -p "$PID" -o pid,pcpu,pmem,rss,vsz,etime
        else
            echo "PID 文件存在但进程已死"
            rm -f "$PID_FILE"
        fi
    else
        echo "$APP_NAME 未运行"
    fi
}

case "$1" in
    start)   start   ;;
    stop)    stop    ;;
    restart) stop; sleep 1; start ;;
    reload)  reload  ;;
    status)  status  ;;
    *)
        echo "用法: $0 {start|stop|restart|reload|status}"
        exit 1
        ;;
esac
```

## Epoll 网络服务器（完整 C 实现）

以下是一个完整的 C 语言 epoll 服务器，包含连接管理、消息缓冲区、协议拆包。可以在 Linux 上用 `gcc -o game_server epoll_server.c -lpthread` 编译。

```c
// epoll_server.c — 完整的游戏服务器骨架
// 功能: TCP 连接管理 + 消息拆包（长度前缀） + 回声处理
// 对比 Raylib: 相当于网络版的 InitWindow/BeginDrawing/EndDrawing 循环

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <fcntl.h>
#include <errno.h>
#include <signal.h>
#include <sys/epoll.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <pthread.h>

// ---- 配置 ----
#define PORT           8888
#define MAX_EVENTS     1024
#define MAX_CLIENTS    2048
#define BUFFER_SIZE    4096
#define HEADER_SIZE    4    // 消息长度前缀占4字节

// ---- 连接结构体 ----
typedef struct {
    int fd;                     // 套接字
    char recv_buf[BUFFER_SIZE]; // 接收缓冲区
    int recv_len;               // 已接收长度
    char send_buf[BUFFER_SIZE]; // 发送缓冲区
    int send_len;               // 待发送长度
    int send_pos;               // 已发送位置
    struct sockaddr_in addr;    // 客户端地址
    int online;                 // 是否在线
    uint64_t last_ping;         // 最后心跳时间（毫秒）
} Connection;

// ---- 全局状态 ----
static Connection g_clients[MAX_CLIENTS];
static int g_epoll_fd;
static int g_server_fd;
static int g_running = 1;

// ---- 工具函数 ----

/// 设置非阻塞
static int set_nonblocking(int fd) {
    int flags = fcntl(fd, F_GETFL, 0);
    if (flags == -1) return -1;
    return fcntl(fd, F_SETFL, flags | O_NONBLOCK);
}

/// 获取连接对象（按 fd）
static Connection* get_connection(int fd) {
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (g_clients[i].fd == fd && g_clients[i].online)
            return &g_clients[i];
    }
    return NULL;
}

/// 分配一个新的连接槽位
static Connection* alloc_connection(int fd, struct sockaddr_in* addr) {
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (!g_clients[i].online) {
            g_clients[i].fd = fd;
            g_clients[i].recv_len = 0;
            g_clients[i].send_len = 0;
            g_clients[i].send_pos = 0;
            g_clients[i].addr = *addr;
            g_clients[i].online = 1;
            g_clients[i].last_ping = 0;
            return &g_clients[i];
        }
    }
    return NULL; // 服务器满了
}

/// 释放连接槽位
static void free_connection(int fd) {
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (g_clients[i].fd == fd) {
            g_clients[i].online = 0;
            g_clients[i].fd = -1;
            break;
        }
    }
}

/// 安全关闭连接
static void close_connection(int fd) {
    if (fd <= 0) return;
    epoll_ctl(g_epoll_fd, EPOLL_CTL_DEL, fd, NULL);
    free_connection(fd);
    close(fd);
    printf("连接关闭: fd=%d\n", fd);
}

// ---- 消息协议 ----
// 协议格式: [4字节消息体长度][消息体]
// 小端序，len 不包含头部自身的4字节

/// 从缓冲区拆包一条完整消息。返回消息体长度，0=还没有完整消息
static int unpack_message(Connection* conn, char** out_msg) {
    if (conn->recv_len < HEADER_SIZE)
        return 0;

    // 读长度前缀
    int body_len = 0;
    memcpy(&body_len, conn->recv_buf, HEADER_SIZE);

    // 检查完整性
    int total_len = HEADER_SIZE + body_len;
    if (conn->recv_len < total_len)
        return 0; // 半包，继续等

    // 消息体指针
    *out_msg = conn->recv_buf + HEADER_SIZE;

    // 移除已完成的消息（将剩余数据前移）
    int remaining = conn->recv_len - total_len;
    if (remaining > 0) {
        memmove(conn->recv_buf, conn->recv_buf + total_len, remaining);
    }
    conn->recv_len = remaining;

    return body_len;
}

/// 封装消息（加长度前缀）
static int pack_message(const char* body, int body_len, char* out_buf, int out_size) {
    if (HEADER_SIZE + body_len > out_size)
        return -1;

    memcpy(out_buf, &body_len, HEADER_SIZE);
    memcpy(out_buf + HEADER_SIZE, body, body_len);
    return HEADER_SIZE + body_len;
}

/// 处理收到的消息（在这里做游戏逻辑分发）
static void handle_message(Connection* conn, const char* msg, int len) {
    if (len <= 0) return;

    // 简单解析消息ID（假设第一条消息的4字节是消息ID）
    // 生产环境用 Protobuf 解析
    if (len < 4) return;

    unsigned int msg_id = 0;
    memcpy(&msg_id, msg, 4);

    printf("收到消息: fd=%d, msg_id=%u, len=%d\n", conn->fd, msg_id, len);

    // 回声测试 — 把收到的消息原样发回
    char response[BUFFER_SIZE];
    int resp_len = pack_message(msg, len, response, BUFFER_SIZE);
    if (resp_len > 0) {
        // 将响应加入发送队列
        memcpy(conn->send_buf, response, resp_len);
        conn->send_len = resp_len;
        conn->send_pos = 0;

        // 通知 epoll 我们要写了
        struct epoll_event ev;
        ev.events = EPOLLIN | EPOLLOUT | EPOLLRDHUP;
        ev.data.fd = conn->fd;
        epoll_ctl(g_epoll_fd, EPOLL_CTL_MOD, conn->fd, &ev);
    }
}

// ---- 核心 IO 逻辑 ----

/// 接收数据
static void handle_read(Connection* conn) {
    while (1) {
        int space = BUFFER_SIZE - conn->recv_len;
        if (space <= 0) {
            // 缓冲区满了，断开防止恶意客户端
            printf("缓冲区溢出: fd=%d\n", conn->fd);
            close_connection(conn->fd);
            return;
        }

        ssize_t n = read(conn->fd, conn->recv_buf + conn->recv_len, space);
        if (n > 0) {
            conn->recv_len += n;
        } else if (n == 0) {
            // 对方关闭连接
            close_connection(conn->fd);
            return;
        } else {
            if (errno == EAGAIN || errno == EWOULDBLOCK)
                break; // 数据读完了
            close_connection(conn->fd);
            return;
        }
    }

    // 拆包并处理所有完整消息
    while (1) {
        char* msg = NULL;
        int msg_len = unpack_message(conn, &msg);
        if (msg_len <= 0) break; // 没有完整消息了

        handle_message(conn, msg, msg_len);
    }
}

/// 发送数据
static void handle_write(Connection* conn) {
    if (conn->send_len <= 0) return;

    while (conn->send_pos < conn->send_len) {
        ssize_t n = write(conn->fd,
            conn->send_buf + conn->send_pos,
            conn->send_len - conn->send_pos);
        if (n > 0) {
            conn->send_pos += n;
        } else {
            if (errno == EAGAIN || errno == EWOULDBLOCK)
                return; // 内核发送缓冲区满了，下次再试
            close_connection(conn->fd);
            return;
        }
    }

    // 全部发完，不再关注 EPOLLOUT
    conn->send_len = 0;
    conn->send_pos = 0;

    struct epoll_event ev;
    ev.events = EPOLLIN | EPOLLRDHUP;
    ev.data.fd = conn->fd;
    epoll_ctl(g_epoll_fd, EPOLL_CTL_MOD, conn->fd, &ev);
}

/// 接受新连接
static void handle_accept() {
    while (1) {
        struct sockaddr_in client_addr;
        socklen_t addr_len = sizeof(client_addr);
        int client_fd = accept(g_server_fd, (struct sockaddr*)&client_addr, &addr_len);
        if (client_fd < 0) {
            if (errno == EAGAIN || errno == EWOULDBLOCK)
                break; // 没有新连接了
            perror("accept");
            break;
        }

        set_nonblocking(client_fd);

        // 分配连接
        Connection* conn = alloc_connection(client_fd, &client_addr);
        if (conn == NULL) {
            printf("服务器已满，拒绝连接: %s\n",
                   inet_ntoa(client_addr.sin_addr));
            close(client_fd);
            break;
        }

        // 加入 epoll
        struct epoll_event ev;
        ev.events = EPOLLIN | EPOLLRDHUP;
        ev.data.fd = client_fd;
        if (epoll_ctl(g_epoll_fd, EPOLL_CTL_ADD, client_fd, &ev) < 0) {
            perror("epoll_ctl add");
            close_connection(client_fd);
            break;
        }

        printf("新连接: fd=%d, %s:%d\n",
               client_fd,
               inet_ntoa(client_addr.sin_addr),
               ntohs(client_addr.sin_port));
    }
}

// ---- SIGINT 处理 ----
static void signal_handler(int sig) {
    if (sig == SIGINT || sig == SIGTERM) {
        printf("\n收到信号 %d，正在关机...\n", sig);
        g_running = 0;
    }
}

// ---- 心跳检查线程（可选） ----
static void* ping_thread(void* arg) {
    (void)arg;
    while (g_running) {
        sleep(5);
        uint64_t now = (uint64_t)time(NULL) * 1000;

        for (int i = 0; i < MAX_CLIENTS; i++) {
            if (!g_clients[i].online) continue;

            // 超过30秒没有ping，断开
            if (now - g_clients[i].last_ping > 30000) {
                printf("心跳超时: fd=%d\n", g_clients[i].fd);
                close_connection(g_clients[i].fd);
            }
        }
    }
    return NULL;
}

// ---- main ----
int main() {
    // 信号处理
    signal(SIGINT, signal_handler);
    signal(SIGTERM, signal_handler);
    signal(SIGPIPE, SIG_IGN); // 防止写已关闭连接时崩溃

    // 创建 socket
    g_server_fd = socket(AF_INET, SOCK_STREAM | SOCK_NONBLOCK, 0);
    if (g_server_fd < 0) {
        perror("socket");
        return 1;
    }

    // 启用端口重用（方便快速重启）
    int opt = 1;
    setsockopt(g_server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    // 绑定
    struct sockaddr_in addr;
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(PORT);

    if (bind(g_server_fd, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
        perror("bind");
        close(g_server_fd);
        return 1;
    }

    if (listen(g_server_fd, SOMAXCONN) < 0) {
        perror("listen");
        close(g_server_fd);
        return 1;
    }

    // 创建 epoll
    g_epoll_fd = epoll_create1(0);
    if (g_epoll_fd < 0) {
        perror("epoll_create1");
        return 1;
    }

    struct epoll_event ev;
    ev.events = EPOLLIN;
    ev.data.fd = g_server_fd;
    epoll_ctl(g_epoll_fd, EPOLL_CTL_ADD, g_server_fd, &ev);

    printf("游戏服务器启动: 0.0.0.0:%d\n", PORT);
    printf("等待客户端连接...\n");

    // 启动心跳线程
    pthread_t ping_tid;
    pthread_create(&ping_tid, NULL, ping_thread, NULL);

    // ---- 主循环（对应 Game Loop） ----
    struct epoll_event events[MAX_EVENTS];

    while (g_running) {
        // 阻塞等待事件（类似 WaitForSeconds）
        int nfds = epoll_wait(g_epoll_fd, events, MAX_EVENTS, 1000);
        if (nfds < 0) {
            if (errno == EINTR) continue; // 被信号中断
            perror("epoll_wait");
            break;
        }

        // 处理所有事件
        for (int i = 0; i < nfds; i++) {
            int fd = events[i].data.fd;

            if (events[i].events & (EPOLLERR | EPOLLHUP | EPOLLRDHUP)) {
                // 连接异常
                close_connection(fd);
                continue;
            }

            if (fd == g_server_fd) {
                handle_accept();
            } else {
                Connection* conn = get_connection(fd);
                if (conn == NULL) {
                    close_connection(fd);
                    continue;
                }

                if (events[i].events & EPOLLIN) {
                    handle_read(conn);
                }

                if (events[i].events & EPOLLOUT) {
                    handle_write(conn);
                }
            }
        }

        // === 在这里可以添加游戏世界 Tick 逻辑 ===
        // if (g_running && game_world) {
        //     game_world_update(game_world, 0.1f);
        // }
    }

    // ---- 清理 ----
    printf("正在关闭服务器...\n");

    // 关闭所有客户端
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (g_clients[i].online) {
            close_connection(g_clients[i].fd);
        }
    }

    close(g_server_fd);
    close(g_epoll_fd);
    printf("服务器已关闭。\n");
    return 0;
}
```

### Epoll 关键概念对比

| 概念 | 说明 | 类比 C++/Raylib |
|------|------|----------------|
| epoll_create1 | 创建 epoll 实例 | 类似 InitWindow |
| epoll_ctl ADD/MOD/DEL | 注册/修改/删除关注的事件 | 类似注册回调 |
| epoll_wait | 等待事件发生 | 类似 WaitForUpdate |
| EPOLLIN | 可读事件 | 有输入 |
| EPOLLOUT | 可写事件 | 缓冲区可写入 |
| EPOLLET | 边缘触发（通知一次，需一次性读完） | — |
| EPOLLRDHUP | 对方关闭连接 | — |

## Redis 数据结构与游戏应用

### 五种基础类型

```bash

# 用途: 存储单个值（验证码、Token、简单配置）

# ======== Sorted Set（有序集合）========
# 用途: 排行榜、优先级队列
# 对比: 类似 std::map<score, id>，但 O(logN) 操作
ZADD leaderboard 15000 "player_1001"
ZADD leaderboard 22000 "player_1002"
ZADD leaderboard 8000  "player_1003"
ZINCRBY leaderboard 500 "player_1001"  # 加分

ZREVRANGE leaderboard 0 9 WITHSCORES  # 前10名
ZRANK leaderboard "player_1001"       # 我的排名(从0开始)
ZSCORE leaderboard "player_1001"      # 我的分数
ZCOUNT leaderboard 0 10000            # 低于10000分的人数
```

### 游戏应用场景代码片段

```c
// --- 用 C 语言操作 Redis（基于 hiredis 库） ---
// gcc -o test_redis test_redis.c -lhiredis

#include <stdio.h>
#include <hiredis/hiredis.h>

// 检查玩家是否在线（用 Set 数据结构）
int is_player_online(redisContext* redis, int uid) {
    redisReply* reply = redisCommand(redis, "SISMEMBER online_users %d", uid);
    int online = (reply && reply->type == REDIS_REPLY_INTEGER) ? reply->integer : 0;
    freeReplyObject(reply);
    return online;
}

// 排行榜操作：玩家通关后加分并返回排名变化
typedef struct {
    int rank;       // 当前排名
    int rank_change; // 排名变化（正=上升）
} RankResult;

RankResult update_leaderboard(redisContext* redis, int uid, int score_gained) {
    RankResult result = {0, 0};

    // 获取加分前排名
    redisReply* before = redisCommand(redis, "ZREVRANK leaderboard player_%d", uid);
    int before_rank = (before && before->type == REDIS_REPLY_INTEGER)
                      ? before->integer + 1 : 0;
    freeReplyObject(before);

    // 加分
    redisReply* after_cmd = redisCommand(redis,
        "ZINCRBY leaderboard %d player_%d", score_gained, uid);
    freeReplyObject(after_cmd);

    // 获取新排名
    redisReply* after = redisCommand(redis, "ZREVRANK leaderboard player_%d", uid);
    int after_rank = (after && after->type == REDIS_REPLY_INTEGER)
                     ? after->integer + 1 : 0;
    freeReplyObject(after);

    result.rank = after_rank;
    result.rank_change = (before_rank > 0 && after_rank > 0)
                         ? before_rank - after_rank : 0;
    return result;
}

int main() {
    redisContext* redis = redisConnect("127.0.0.1", 6379);
    if (redis == NULL || redis->err) {
        printf("Redis 连接失败: %s\n", redis ? redis->errstr : "null");
        return 1;
    }

    // 测试排行榜
    RankResult r = update_leaderboard(redis, 1001, 500);
    printf("玩家 1001 当前排名: %d (上升 %d 名)\n", r.rank, r.rank_change);

    redisFree(redis);
    return 0;
}
```

```
// │                   状态同步 vs 帧同步                        │
// ├──────────────┬────────────────────────┬───────────────────┤
// │              │      状态同步           │      帧同步        │
// ├──────────────┼────────────────────────┼───────────────────┤

// │ 服务器负载   │ 高（每帧推送状态）       │ 低（只传输入）     │
// │ 防外挂       │ 好（服务器计算）         │ 差（客户端计算）   │
// │ 实现难度     │ 中等                    │ 高（需确定性）     │
// │ 网络要求     │ 一般                    │ 高（需稳定)       │
// │ 回放         │ 难                      │ 容易（录输入即可） │
```

```bash
# analyze_log.py — 分析游戏日志，计算关键指标

import re
import sys
from collections import Counter
from datetime import datetime

LOG_PATTERN = re.compile(
    r"(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})\s+"
    r"\[(\w+)\]\s+"
    r"(?:\[uid=(\d+)\])?\s*"
    r"(.*)"

    hourly_logins = Counter()
    errors = []

            if not m:

            timestamp, level, uid, msg = m.groups()
            level_counts[level] += 1

            if level == "ERROR":
                errors.append((timestamp, msg))

            # 统计登录（假设登录日志含 "LOGIN" 关键词）

    print(f"时间范围: ... 至 ...")
    print(f"总行数: {sum(level_counts.values())}")
    print(f"级别分布: {dict(level_counts)}")

    if errors:
        print(f"\n错误数量: {len(errors)}")
        for t, msg in errors[-5:]:  # 最近5个错误
            print(f"  [{t}] {msg}")

    print("\n按小时登录数 (Top 5):")
    for hour, count in hourly_logins.most_common(5):
```

```c
// bench_client.c — 简易压测客户端
// 创建多个 TCP 连接发送假消息
// 编译: gcc -o bench bench_client.c -lpthread -lrt

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <pthread.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <time.h>

typedef struct {
    const char* server_ip;
    int server_port;
    int num_connections;
    int duration_secs;
    int msg_size;
} BenchConfig;

typedef struct {
    int fd;
    int sent_count;
    int recv_count;
    int errors;
} ConnStats;

static int g_running = 1;

/// 单个连接的压测线程
static void* bench_thread(void* arg) {
    ConnStats* stats = (ConnStats*)arg;

    // 创建连接
    stats->fd = socket(AF_INET, SOCK_STREAM, 0);
    if (stats->fd < 0) {
        stats->errors++;
        return NULL;
    }

    struct sockaddr_in addr;
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = inet_addr("127.0.0.1");
    addr.sin_port = htons(8888);

    if (connect(stats->fd, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
        perror("connect");
        stats->errors++;
        close(stats->fd);
        return NULL;
    }

    // 准备测试消息
    char send_buf[256];
    int send_len = 8; // 4字节msgId + 4字节seq
    *(int*)send_buf = htonl(1);       // msg_id = 1
    *(int*)(send_buf + 4) = htonl(0); // seq

    char recv_buf[4096];

    // 按时间循环发送
    time_t end_time = time(NULL) + 10; // 10秒测试
    while (g_running && time(NULL) < end_time) {
        // 发送消息（长度前缀 + 消息体）
        uint32_t msg_len = send_len;
        uint32_t net_len = htonl(msg_len);
        struct iovec iov[2];
        iov[0].iov_base = &net_len;
        iov[0].iov_len = 4;
        iov[1].iov_base = send_buf;
        iov[1].iov_len = send_len;

        ssize_t n = writev(stats->fd, iov, 2);
        if (n <= 0) { stats->errors++; break; }
        stats->sent_count++;

        // 读取响应
        n = read(stats->fd, recv_buf, sizeof(recv_buf));
        if (n > 0) stats->recv_count++;
        else if (n == 0) break;

        // 稍微等待避免打满
        usleep(1000); // 1ms
    }

    close(stats->fd);
    return NULL;
}

int main(int argc, char** argv) {
    int connections = 100;
    int duration = 10;

    if (argc > 1) connections = atoi(argv[1]);
    if (argc > 2) duration = atoi(argv[2]);

    printf("=== 压力测试 ===\n");
    printf("连接数: %d, 持续时间: %d秒\n", connections, duration);
    printf("目标: 127.0.0.1:8888\n\n");

    pthread_t* threads = malloc(connections * sizeof(pthread_t));
    ConnStats* stats = calloc(connections, sizeof(ConnStats));

    // 启动所有连接
    for (int i = 0; i < connections; i++) {
        pthread_create(&threads[i], NULL, bench_thread, &stats[i]);
    }

    // 等待完成
    for (int i = 0; i < connections; i++) {
        pthread_join(threads[i], NULL);
    }

    // 汇总统计
    int total_sent = 0, total_recv = 0, total_err = 0;
    for (int i = 0; i < connections; i++) {
        total_sent += stats[i].sent_count;
        total_recv += stats[i].recv_count;
        total_err += stats[i].errors;
    }

    printf("\n=== 测试结果 ===\n");
    printf("成功连接: %d/%d\n", connections - total_err, connections);
    printf("总发送: %d, 总接收: %d\n", total_sent, total_recv);
    printf("吞吐量: %.0f msg/sec\n", (double)total_recv / duration);
    printf("错误数: %d\n", total_err);

    free(threads);
    free(stats);
    return 0;
}
```

```
// 1. 网络层调优
//   □ 使用 epoll ET（边缘触发）而非 LT（水平触发）
//   □ 接收缓冲区 64KB+，防止半包频繁触发
//   □ 合并小包发送（Nagle 算法开/关根据场景）
//   □ 消息池化（避免每次发送 malloc/free）

// 2. 内存调优

//   □ 预分配内存池（固定大小对象池，类似 Unity 对象池）
//   □ 零拷贝：尽量避免不必要的 memcpy
//     Example: 直接使用 recv 缓冲区解析，不额外拷贝
//   □ 减少锁竞争：每个连接独立缓冲区，主线程单线程处理

// 3. 数据库调优
//   □ Redis 用 pipeline（减少网络往返）
//     redisCommand(redis, "PIPELINE");
//     for (...) { redisCommand(redis, "SET key%d val%d", i, i); }

//   □ 批量写入（每 5 秒 flush 一次，而非每次操作写一次）

//   □ 分帧处理（每帧最多处理 N 个玩家的 AI）
//   □ 场景分线（超过 500 人自动开新线）
```
